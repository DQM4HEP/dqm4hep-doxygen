152  this->parseCmdLine(argc, argv);
153  }
154  catch(...) {
155  dqm_error( "Application::init(): failed to parse cmd line !" );
156  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
157  }
158 
159  if(noServer()) {
160  // configure logger
161  m_logger = core::Logger::createLogger(this->type() + ":" + this->name(), {
162  core::Logger::coloredConsole()
163  });
164  }
165  else {
166  // configure logger
167  m_logger = core::Logger::createLogger(this->type() + ":" + this->name(), {
168  core::Logger::coloredConsole(),
169  RemoteLogger::make_shared()
170  });
171  }
172  core::Logger::setMainLogger(m_logger->name());
173  m_logger->set_level(m_logLevel);
174 
175  if(not noServer()) {
176  // configure server
177  dqm_debug( "Creating server ..." );
178  m_server = std::make_shared<net::Server>(OnlineRoutes::Application::serverName(this->type(), this->name()));
179  m_pAppStateService = m_server->createService(OnlineRoutes::Application::state(this->type(), this->name()));
180 
181  if(m_statsEnabled) {
182  dqm_debug( "Creating internal app stats ..." );
183  createInternalStats();
184  }
185  m_server->onClientExit().connect(this, &Application::sendClientExitEvent);
186  }
187 
188  try {
189  dqm_info( "User init ..." );
190  this->onInit();
191  }
192  catch(...) {
193  dqm_error( "Application::init(): failed to initialize the app !" );
194  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);

203  int Application::exec() {
204  m_running = true;
205  if(not noServer()) {
206  m_server->start();
207  sleep(1);
208  }
209  this->setState("Running");
210 
211  try {
212  dqm_info( "Application on start ..." );
213  this->onStart();
214  }
215  catch(...) {
216  dqm_error( "Application::init(): failed to start the app !" );
217  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
218  }
219 
220  int returnCode(0);
221 
222  try {
223  if(statsEnabled() and not noServer()) {
224  // send statistics every 5 seconds
225  dqm_info( "Statistics enabled. Creating timer (5 secs) ..." );
226  m_appStatTimer = createTimer();
227  m_appStatTimer->setInterval(5000); // 5 seconds
228  m_appStatTimer->setSingleShot(false);
229  m_appStatTimer->onTimeout().connect(this, &Application::sendAppStats);
230  m_appStatTimer->start();
231  }
232  m_eventLoop.connectOnEvent(this, &Application::onEvent);
233  dqm_info( "Starting event loop ..." );
234  returnCode = m_eventLoop.exec();
235  dqm_info( "Exiting event loop ..." );
236  m_eventLoop.disconnectOnEvent(this);
237  if(not noServer()){
238  m_server->onClientExit().disconnect();
239  }
240  this->onStop();
241  }
242  catch(...) {
243  dqm_error( "Application::init(): failed to start the app !" );
244  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
245  }
246  std::stringstream appState; appState << "Exiting (" << returnCode << ")";
247  this->setState(appState.str());
248  m_running = false;
249 
250  return returnCode;
251  }

346  void Application::sendAppStats() {
347  core::MemoryStats memInfo;
348  core::memStats(memInfo);
349  core::procStats(m_stats);
350 
351  long procVm = m_stats.vm/1024.; // in MB
352  long procRss = m_stats.rss/1024.; // in MB
353 
354  dqm_debug("Sending internal app stats ..." );
355  sendStat("VmProc", procVm);
356  sendStat("VmTotal", (procVm/(memInfo.vmTot*1.))*100.);
357  sendStat("VmInUse", (procVm/(memInfo.vmUsed*1.))*100.);
358 
359  sendStat("RSSProc", procRss);
360  sendStat("RSSTotal", (procRss/(memInfo.rssTot*1.))*100.);
361  sendStat("RSSInUse", (procRss/(memInfo.rssUsed*1.))*100.);
362 
363  sendStat("CpuUser", m_stats.cpuUser);
364  sendStat("CpuSys", m_stats.cpuSys);
365  sendStat("CpuTotal", m_stats.cpuTot);
366 
367  auto tm_time = localtime(&m_stats.lastPollTime.tv_sec);
368  char date_buf[100];
369  std::strftime(date_buf, sizeof(date_buf), "%Y-%m-%d %H:%M:%S", tm_time);
370 
371  sendStat("LastUpdate", date_buf );
372  dqm_debug( "Sending internal app stats ... OK" );
373  }

435  void Application::NetworkHandler::postServiceContent(const net::Buffer &buffer) {
436  int count = m_eventLoop.count([this](AppEvent* ptr){
437  ServiceUpdateEvent *evt = dynamic_cast<ServiceUpdateEvent*>(ptr);
438  return (evt && evt->serviceName() == this->m_name);
439  });
440 
441  if(count >= m_maxNEvents) {
442  dqm_debug( "NetworkHandler::postServiceContent(): maximum of posted service updates reached ({0}) !", m_maxNEvents );
443  return;
444  }
445 
446  // copy buffer content
447  std::string content;
448  content.assign(buffer.begin(), buffer.size());
449 
450  // create new buffer model
451  auto bufferModel = buffer.createModel<std::string>();
452  bufferModel->move(std::move(content));
453 
454  // create the event to post, pass the copied buffer in ctor
455  ServiceUpdateEvent *pEvent = new ServiceUpdateEvent(m_name, bufferModel);
456  pEvent->setPriority(m_priority);
457 
458  // post the event !
459  m_eventLoop.postEvent(pEvent);
460  }
