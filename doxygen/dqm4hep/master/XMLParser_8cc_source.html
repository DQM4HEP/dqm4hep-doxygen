231  void XMLParser::resolveIncludes(const std::string &referencePath, TiXmlNode *node) {
232  TiXmlElement *child = node->FirstChildElement();
233 
234  while (true) {
235  if (child == nullptr)
236  break;
237 
238  if (child->Value() != std::string("include")) {
239  this->resolveIncludes(referencePath, child);
240  child = child->NextSiblingElement();
241  continue;
242  }
243 
244  // process the include and returns the parsed document
245  TiXmlDocument doc;
246  this->resolveInclude(referencePath, child, doc);
247 
248  TiXmlElement *rootInclude = doc.RootElement();
249 
250  if (std::string(rootInclude->Value()) != "dqm4hep") {
251  dqm_error("XMLParser::resolveIncludes(): no <dqm4hep> root element in included file '{0}'", doc.Value());
252  throw StatusCodeException(STATUS_CODE_FAILURE);
253  }
254 
255  // add them to the xml tree
256  TiXmlNode *includeAfter(child);
257 
258  for (TiXmlElement *includeElement = rootInclude->FirstChildElement(); includeElement != nullptr;
259  includeElement = includeElement->NextSiblingElement())
260  includeAfter = node->InsertAfterChild(includeAfter, *includeElement);
261 
262  // case where at least one element has been inserted from the include file
263  if(includeAfter != child) {
264  node->RemoveChild(child);
265  child = includeAfter->NextSiblingElement();
266  }
267  // case where nothing was inserted
268  else {
269  dqm_debug( "XMLParser::resolveIncludes(): No element in included file" );
270  // Find the previous element to point to if any
271  TiXmlElement *previousElement(nullptr);
272  TiXmlNode *previousNode(child);
273  while(nullptr != previousNode->PreviousSibling()) {
274  if(nullptr != previousNode->PreviousSibling()->ToElement()) {
275  previousElement = previousNode->PreviousSibling()->ToElement();
276  break;
277  }
278  else {
279  previousNode = previousNode->PreviousSibling();
280  }
281  }
282  // node->RemoveChild(child);
283  // if a previous element is available,
284  // continue the while from it
285  if(nullptr != previousElement) {
286  dqm_debug( "XMLParser::resolveIncludes(): Previous element found" );
287  node->RemoveChild(child);
288  child = previousElement;
289  }
290  // if no previous element is available,
291  // restart from the first child of input
292  else {
293  dqm_debug( "XMLParser::resolveIncludes(): No previous element" );
294  node->RemoveChild(child);
295  child = node->FirstChildElement();
296  }
297  }
298 
299  }

420  void XMLParser::performConstantReplacement(std::string &value) {
421  size_t pos = value.find("${");
422  std::string original(value);
423 
424  while (pos != std::string::npos) {
425  size_t pos2 = value.find_first_of("}", pos + 2);
426 
427  if (pos2 == std::string::npos) {
428  dqm_error("XMLParser::performConstantReplacement(): couldn't parse constant value !");
429  throw StatusCodeException(STATUS_CODE_FAILURE);
430  }
431 
432  const std::string key(value.substr(pos + 2, pos2 - pos - 2));
433  auto findConstant = m_constants.find(key);
434  const std::string replacementValue(findConstant != m_constants.end() ? findConstant->second : "");
435 
436  if (replacementValue.empty()) {
437  dqm_error("XMLParser::performConstantReplacement(): constant \"{0}\" not found in available constants !",
438  key);
439  throw StatusCodeException(STATUS_CODE_FAILURE);
440  }
441 
442  value.replace(pos, (pos2 + 1 - pos), replacementValue);
443  pos2 = pos + replacementValue.size(); // new end position after replace
444  pos = value.find("${", pos2); // find next possible key to replace
445  }
446  }

484  // replace constants in all attributes
485  for (TiXmlAttribute *attribute = element->FirstAttribute(); attribute != nullptr; attribute = attribute->Next()) {
486  std::string value(attribute->ValueStr());

489 
490  if (this->allowEnvVariables())
491  this->performEnvVariableReplacement(value);
492 
493  attribute->SetValue(value);

495 
496  for (TiXmlNode *node = element->FirstChild(); node != nullptr; node = node->NextSibling()) {
497  if (node->Type() == TiXmlNode::TINYXML_ELEMENT) {
498  TiXmlElement *child = node->ToElement();
499  this->resolveConstantsInElement(child);

501  TiXmlText *child = node->ToText();
502  std::string value(child->ValueStr());
503 
504  this->performConstantReplacement(value);

507  this->performEnvVariableReplacement(value);

509  child->SetValue(value);
510  }

550  void XMLParser::resolveAllDbSelect(TiXmlElement *pXmlElement) {
551  if (m_databases.empty())
552  return;
553 
554  for (TiXmlElement *element = pXmlElement->FirstChildElement(); element != nullptr;
555  element = element->NextSiblingElement()) {
556  if (std::string(element->Value()) == "select") {
557  std::string dbID = element->Attribute("db") != nullptr ? element->Attribute("db") : "";
558  std::string dbTable = element->Attribute("table") != nullptr ? element->Attribute("table") : "";
559 
560  if (dbID.empty() || dbTable.empty()) {
561  dqm_error("<select> element, row {0}, column {1}: invalid/missing attributes 'db' or 'table'",
562  element->Row(), element->Column());
563  throw StatusCodeException(STATUS_CODE_FAILURE);
564  }
565 
566  auto findIter = m_databases.find(dbID);
567 
568  if (m_databases.end() == findIter) {
569  dqm_error("<select> element, row {0}, column {1}: db id '{2}' not found !", element->Row(),
570  element->Column(), dbID);
571  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
572  }
573 
574  std::shared_ptr<DBInterface> dbInterface(findIter->second);
575  StringMap parameterMap;
576  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, dbInterface->getTableParameters(dbTable, parameterMap));
577 
578  TiXmlNode *pInsertAfter(element), *pParentElement(element->Parent()->ToElement());
579 
580  for (auto parameterIter : parameterMap) {
581  dqm_info("Selected parameter '{0}' = '{1}'", parameterIter.first, parameterIter.second);
582 
583  auto pParameterElement = new TiXmlElement("parameter");
584  pParameterElement->SetAttribute("name", parameterIter.first);
585  pParameterElement->SetAttribute("value", parameterIter.second);
586  pInsertAfter = pParentElement->InsertAfterChild(pInsertAfter, *pParameterElement);
587  delete pParameterElement;
588  }
589 
590  TiXmlElement *previousElement(nullptr);
591  TiXmlNode *previousNode(element);
592 
593  while(nullptr != previousNode->PreviousSibling()) {
594  if(nullptr != previousNode->PreviousSibling()->ToElement()) {
595  previousElement = previousNode->PreviousSibling()->ToElement();
596  break;
597  }
598  else {
599  previousNode = previousNode->PreviousSibling();
600  }
601  }
602  // if a previous element is available,
603  // continue the loop from it
604  if(nullptr != previousElement) {
605  pParentElement->RemoveChild(element);
606  element = previousElement;
607  }
608  // if no previous element is available,
609  // restart from the first child of input
610  else {
611  pParentElement->RemoveChild(element);
612  element = pParentElement->FirstChildElement();
613  }
614  } else {
615  // recursive call on childs !
616  this->resolveAllDbSelect(element);
617  }
618  }
619  }

630  void XMLParser::resolveForLoops(TiXmlElement *element) {
631  // first, resolve for loops in child elements
632  for(auto child = element->FirstChildElement(); nullptr != child ; child = child->NextSiblingElement()) {
633  resolveForLoops(child);
634  }
635  // if loop element, resolve it
636  if(element->Value() == std::string("for")) {
637  auto parent = element->Parent();
638  if(nullptr == parent) {
639  return;
640  }
641  int begin(0), end(0), increment(0);
642  std::string loopId;
643  if(TIXML_SUCCESS != element->QueryIntAttribute("begin", &begin)) {
644  dqm_error( "XMLParser::resolveForLoops: <for> element with invalid 'begin' attribute !" );
645  throw StatusCodeException(STATUS_CODE_FAILURE);
646  }
647  if(TIXML_SUCCESS != element->QueryIntAttribute("end", &end)) {
648  dqm_error( "XMLParser::resolveForLoops: <for> element with invalid 'end' attribute !" );
649  throw StatusCodeException(STATUS_CODE_FAILURE);
650  }
651  if(TIXML_SUCCESS != element->QueryStringAttribute("id", &loopId) or loopId.empty()) {
652  dqm_error( "XMLParser::resolveForLoops: <for> element with invalid 'id' attribute !" );
653  throw StatusCodeException(STATUS_CODE_FAILURE);
654  }
655  // optional attribute
656  element->QueryIntAttribute("increment", &increment);
657  const bool forwardLoop(begin < end);
658  if((forwardLoop and increment < 0) or (not forwardLoop and increment > 0)) {
659  dqm_error( "XMLParser::resolveForLoops: <for> element with infinite loop detected !" );
660  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
661  }
662  for(int i=begin ; i<=end ; i+=increment) {
663  for(auto child = element->FirstChild() ; nullptr != child ; child = child->NextSibling()) {
664  auto cloneChild = child->Clone();
665  if(nullptr != cloneChild) {
666  resolveForLoop(cloneChild, loopId, i);
667  parent->InsertBeforeChild(element, *cloneChild);
668  delete cloneChild;
669  }
670  }
671  }
672  // remove the element itself from the tree
673  parent->RemoveChild(element);
674  }
675  }

679  void XMLParser::resolveForLoop(TiXmlNode *node, const std::string &id, int value) {
680  std::string loopIDStr = "$FOR{" + id + "}";
681  std::string valueStr = typeToString(value);
682  if(node->Type() == TiXmlNode::TINYXML_ELEMENT) {
683  TiXmlElement *nodeElement = node->ToElement();
684  // replace occurences in all attributes
685  for(auto attr = nodeElement->FirstAttribute() ; nullptr != attr ; attr = attr->Next()) {
686  std::string attrValue(attr->ValueStr());
687  size_t pos = attrValue.find(loopIDStr, 0);
688  while(pos != std::string::npos) {
689  attrValue.replace(pos, loopIDStr.size(), valueStr);
690  pos = attrValue.find(loopIDStr, pos+valueStr.size());
691  }
692  attr->SetValue(attrValue);
693  }
694  // make it recursive
695  for(auto child = nodeElement->FirstChild() ; nullptr != child ; child = child->NextSibling()) {
696  resolveForLoop(child, id, value);
697  }
698  }
699  else if((node->Type() == TiXmlNode::TINYXML_COMMENT) or (node->Type() == TiXmlNode::TINYXML_TEXT)) {
700  std::string nodeValue = node->ValueStr();
701  size_t pos = nodeValue.find(loopIDStr, 0);
702  while(pos != std::string::npos) {
703  nodeValue.replace(pos, loopIDStr.size(), valueStr);
704  pos = nodeValue.find(loopIDStr, pos+valueStr.size());
705  }
706  node->SetValue(nodeValue);
707  }
708  }
