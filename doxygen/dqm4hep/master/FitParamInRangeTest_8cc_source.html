197  StatusCode FitParamInRangeTest::readSettings(const TiXmlHandle xmlHandle) {
198  UInt_t nParameters = 0;
199  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::readParameter(xmlHandle,
200  "FitFormula", m_fitFormula, [&nParameters,this](const std::string &value){
201  if(value.empty()) {
202  return false;
203  }
204  std::stringstream formulaName;
205  formulaName << "Formula_RS_" << (void*)this;
206  TFormula formula(formulaName.str().c_str(), value.c_str(), true);
207  nParameters = formula.GetNpar();
208  if(0 == nParameters) {
209  return false;
210  }
211  return true;
212  }));
213

216  return (values.size() <= nParameters);

219  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::readParameter(xmlHandle,

251  void FitParamInRangeTest::userRun(MonitorElement* monitorElement, QualityTestReport &report) {
252  report.m_extraInfos["formula"] = m_fitFormula;
253  // check monitor element first
254  if(not checkElement(monitorElement)) {
255  report.m_message = "Object attached to monitor element is not a TH1, TGraph or TGraph2D !";
256  report.m_quality = 0.f;
257  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
258  }
259  // create the function
260  TF1 *fitFunction = createFunction();
261  std::unique_ptr<TF1> up(fitFunction);
262  // check monitor element <-> function matching
263  if(not checkElement(monitorElement, fitFunction, report)) {
264  throw StatusCodeException(STATUS_CODE_FAILURE);
265  }
266  // perform fit
267  TFitResultPtr fitResult = performFit(monitorElement, fitFunction, report);
268  up.release();
269  // perform range check
270  performRangeTest(fitResult, report);
271  }

275  TF1* FitParamInRangeTest::createFunction() const {
276  // Create our function
277  std::stringstream formulaName;
278  formulaName << "FitFormula_" << (void*)this;
279  TFormula formula(formulaName.str().c_str(), m_fitFormula.c_str(), true);
280  TF1 *fitFunction = nullptr;
281  if(formula.GetNdim() == 1) {
282  dqm_debug( "Created function 1D" );
283  fitFunction = new TF1(formulaName.str().c_str(), m_fitFormula.c_str(), 0, 1);
284  }
285  else if(formula.GetNdim() == 2) {
286  dqm_debug( "Created function 2D" );
287  fitFunction = new TF2(formulaName.str().c_str(), m_fitFormula.c_str(), 0, 1, 0, 1);
288  }
289  else if(formula.GetNdim() == 3) {
290  dqm_debug( "Created function 3D" );
291  fitFunction = new TF3(formulaName.str().c_str(), m_fitFormula.c_str(), 0, 1, 0, 1, 0, 1);
292  }
293  else {
294  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
295  }
296  // set guess parameters
297  dqm_debug( "Settings guess parameters ..." );
298  for(unsigned int i=0 ; i<m_guessParameters.size() ; ++i) {
299  fitFunction->SetParameter(i, m_guessParameters.at(i));
300  }
301  dqm_debug( "Settings guess parameters ... OK" );
302  // set function range
303  dqm_debug( "Settings function range ..." );
304  Double_t functionRange[6] = {0};
305  for(unsigned int i=0 ; i<6 ; ++i) {
306  if(i < m_functionRange.size()) {
307  functionRange[i] = m_functionRange.at(i);
308  }
309  else {
310  functionRange[i] = 0.;
311  }
312  }
313  if(not m_functionRange.empty()) {
314  dqm_debug( "Calling SetRange(...)" );
315  fitFunction->SetRange(
316  functionRange[0], functionRange[2], functionRange[4],
317  functionRange[1], functionRange[3], functionRange[5]
318  );
319  }
320  dqm_debug( "Settings function range ... OK" );
321  return fitFunction;
322  }

351  report.m_message = "TGraph2D attached to monitor element and fit formula have different dimensions !";
352  report.m_quality = 0.f;
353  return false;

383  TFitResultPtr FitParamInRangeTest::performFit(MonitorElement* monitorElement, TF1 *fitFunction, QReport &report) const {
384  TH1 *histogram = monitorElement->objectTo<TH1>();
385  TGraph *graph = monitorElement->objectTo<TGraph>();
386  TGraph2D *graph2D = monitorElement->objectTo<TGraph2D>();
387  // get the fit options
388  bool isHistogram = (nullptr != monitorElement->objectTo<TH1>());
389  bool withSumW2 = false;
390  if(isHistogram)
391  withSumW2 = (0 != monitorElement->objectTo<TH1>()->GetSumw2N());
392  std::string fitOptions = getFitOptions(isHistogram, withSumW2);
393  report.m_extraInfos["fit.opt"] = fitOptions;
394  // perform fit
395  TFitResultPtr fitResult;
396  if(nullptr != histogram) {
397  dqm_debug( "Fitting histogram" );
398  fitResult = histogram->Fit(fitFunction, fitOptions.c_str(), "");
399  }
400  else if(nullptr != graph) {
401  dqm_debug( "Fitting graph" );
402  fitResult = graph->Fit(fitFunction, fitOptions.c_str(), "");
403  }
404  else if(nullptr != graph2D) {
405  dqm_debug( "Fitting graph2D" );
406  TF2 *fitFunction2D = (TF2*)fitFunction;
407  fitResult = graph2D->Fit(fitFunction2D, fitOptions.c_str(), "");
408  }
409  dqm_debug( "Fitting done ..." );
410  if(not fitResult->IsValid() or fitResult->IsEmpty()) {
411  report.m_message = "Fit operation returned error !";
412  report.m_quality = 0.f;
413  throw StatusCodeException(STATUS_CODE_FAILURE);
414  }
415  return fitResult;
416  }

420  void FitParamInRangeTest::performRangeTest(TFitResultPtr fitResult, QReport &report) const {
421  const double testParameter = fitResult->GetParams()[m_testParameter];
422  const double testParameterError = fitResult->GetErrors()[m_testParameter];
423  const double testParameterMinus = testParameter - testParameterError;
424  const double testParameterPlus = testParameter + testParameterError;
425 
426  if(withinRange(testParameter, m_deviationLower, m_deviationUpper)) {
427  report.m_message = "Fit parameter within range";
428  report.m_quality = 1.f; // TODO scale me correctly !
429  }
430  else if(testParameterPlus > m_deviationLower or testParameterMinus < m_deviationUpper) {
431  report.m_message = "Fit parameter not within range, but by using the fit parameter error, the value is within the expected range";
432  report.m_quality = (this->errorLimit() + this->warningLimit())/2.f; // TODO scale me correctly !
433  }
434  else {
435  report.m_message = "Fit parameter not within range !";
436  report.m_quality = this->errorLimit() / 2.f; // TODO scale me correctly !
437  }
438  report.m_extraInfos["fit.val"] = testParameter;
439  report.m_extraInfos["fit.err"] = testParameterError;
440  report.m_extraInfos["fit.prob"] = fitResult->Prob();
441  report.m_extraInfos["fit.status"] = fitResult->Status();
442  report.m_extraInfos["fit.ncalls"] = fitResult->NCalls();
443  report.m_extraInfos["fit.mintype"] = fitResult->MinimizerType();
444  report.m_extraInfos["fit.minfcn"] = fitResult->MinFcnValue();
445  report.m_extraInfos["fit.edm"] = fitResult->Edm();
446  report.m_extraInfos["fit.chi2"] = fitResult->Chi2();
447  }
