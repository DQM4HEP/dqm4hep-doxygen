92  inline StatusCode Storage<T>::mkdir(const std::string &dirName) {
93  if (dirName.empty())
94  return STATUS_CODE_INVALID_PARAMETER;
95 
96  Path path(dirName);
97 
98  if (!path.isValid())
99  return STATUS_CODE_INVALID_PARAMETER;
100 
101  DirectoryPtr directory = !path.isRelative() ? m_rootDirectory : m_currentDirectory;
102  StringVector directoryList = path.getSplitPath();
103 
104  for (auto iter = directoryList.begin(), endIter = directoryList.end(); endIter != iter; ++iter) {
105  std::string dname = *iter;
106 
107  if (dname == ".")
108  continue;
109 
110  if (dname == "..") {
111  if (directory == m_rootDirectory)
112  return STATUS_CODE_FAILURE;
113 
114  directory = directory->parent();
115  continue;
116  }
117 
118  // if sub dir doesn't exists, create it
119  if (!directory->hasChild(dname))
120  directory->mkdir(dname);
121 
122  // navigate forward
123  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, directory->find(dname, directory));
124  }
125 
126  return STATUS_CODE_SUCCESS;
127  }

210  inline StatusCode Storage<T>::rmdir(const std::string &dirName) {
211  if (dirName.empty())
212  return STATUS_CODE_NOT_ALLOWED;
213 
214  DirectoryPtr directory = nullptr;
215  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->find(dirName, directory));
216 
217  if (directory == m_rootDirectory)
218  return STATUS_CODE_NOT_ALLOWED;
219 
220  std::string fullPathDirName = directory->fullPath().getPath();
221  std::string currentFullPathDirName = m_currentDirectory->fullPath().getPath();
222  size_t pos = currentFullPathDirName.find(fullPathDirName);
223 
224  // this mean that the directory that we try
225  // to remove is a parent of the current one.
226  if (pos == 0 || pos != std::string::npos)
227  return STATUS_CODE_FAILURE;
228 
229  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, directory->parent()->rmdir(directory->name()));
230 
231  return STATUS_CODE_SUCCESS;
232  }
