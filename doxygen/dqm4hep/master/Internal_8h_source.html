345  inline void tokenize(const std::string &inputString, std::vector<T> &tokens, const std::string &delimiter = " ") {
346  std::string::size_type lastPos = inputString.find_first_not_of(delimiter, 0);
347  std::string::size_type pos = inputString.find_first_of(delimiter, lastPos);
348 
349  while ((std::string::npos != pos) || (std::string::npos != lastPos)) {
350  T value;
351  if(not stringToType(inputString.substr(lastPos, pos - lastPos), value)) {
352  return;
353  }
354  tokens.push_back(value);
355  lastPos = inputString.find_first_not_of(delimiter, pos);
356  pos = inputString.find_first_of(delimiter, lastPos);
357  }
358  }
