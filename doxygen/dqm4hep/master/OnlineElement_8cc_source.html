222  core::StatusCode OnlineElement::write(TBuffer &buffer) const {
223  // write base
224  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::MonitorElement::write(buffer));
225  // write properties
226  buffer.WriteInt(m_runNumber);
227  buffer.WriteStdString(&m_collectorName);
228  buffer.WriteStdString(&m_moduleName);
229  buffer.WriteStdString(&m_description);
230  core::json reports = {};
231  for(auto report : m_reports) {
232  core::json jreport;
233  report.second.toJson(jreport);
234  reports[report.first] = jreport;
235  }
236  const std::string qreportStr = reports.dump();
237  buffer.WriteStdString(&qreportStr);
238  return core::STATUS_CODE_SUCCESS;

243  core::StatusCode OnlineElement::read(TBuffer &buffer) {
244  reset(false);
245  // read base
246  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::MonitorElement::read(buffer));
247  // read properties
248  std::string qreportStr;
249  buffer.ReadInt(m_runNumber);
250  buffer.ReadStdString(&m_collectorName);
251  buffer.ReadStdString(&m_moduleName);

254  auto reports = core::json::parse(qreportStr);
255  for(auto it = reports.begin() ; it != reports.end() ; it++) {
256  core::QReport report; report.fromJson(it.value());
