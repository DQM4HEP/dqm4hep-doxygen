28  void RootStyle::applyTo(TObject *object) const {
29  if(nullptr == object) {
30  return;
31  }
32  TAttMarker *markerAtt = dynamic_cast<TAttMarker*>(object);
33  if(nullptr != markerAtt) {
34  markerAtt->SetMarkerColor(m_markerColor);
35  markerAtt->SetMarkerStyle(m_markerStyle);
36  markerAtt->SetMarkerSize(m_markerSize);

38  TAttLine *lineAtt = dynamic_cast<TAttLine*>(object);

40  lineAtt->SetLineColor(m_lineColor);

44  TAttFill *fillAtt = dynamic_cast<TAttFill*>(object);
45  if(nullptr != fillAtt) {
