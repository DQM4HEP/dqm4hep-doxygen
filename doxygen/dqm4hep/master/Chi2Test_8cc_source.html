108  void Chi2Test::userRun(MonitorElement* pMonitorElement, QualityTestReport &report) {
109  const bool hasObject = (pMonitorElement->object() != nullptr);
110  const bool hasReference = (pMonitorElement->reference() != nullptr);
111 
112  if (!hasObject) {
113  report.m_message = "No object attached to monitor element";
114  report.m_quality = 0.f;
115  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
116  }
117 
118  if (!hasReference) {
119  report.m_message = "No reference attached to monitor element";
120  report.m_quality = 0.f;
121  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
122  }
123 
124  const bool isObjHistogram = pMonitorElement->object()->InheritsFrom("TH1");
125  const bool isRefHistogram = pMonitorElement->reference()->InheritsFrom("TH1");
126 
127  if (!isObjHistogram or !isRefHistogram) {
128  report.m_message = "Both the object and the reference must be a TH1";
129  report.m_quality = 0.f;
130  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
131  }
132 
133  TH1* pHistogram = pMonitorElement->objectTo<TH1>();
134  TH1* pReferenceHistogram = pMonitorElement->referenceTo<TH1>();
135  std::string options = Chi2Test::getTestOptions(m_comparisonType);
136 
137  report.m_quality = pHistogram->Chi2Test(pReferenceHistogram, options.c_str());
138  }
