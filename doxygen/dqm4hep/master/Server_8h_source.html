283  void (Controller::*function)(const Buffer &request, Buffer &response)) {
284  auto findIter = m_requestHandlerMap.find(rname);

286  if (findIter != m_requestHandlerMap.end())
287  throw std::runtime_error("Server::createRequestHandler(): request handler '" + rname +
288  "' already exists in this client");
289 
290  if (Server::requestHandlerAlreadyRunning(rname))
291  throw std::runtime_error("Server::createRequestHandler(): request handler '" + rname +
292  "' already running on network");
293 
294  // first insert nullptr, then create request handler
295  auto inserted = m_requestHandlerMap.insert(RequestHandlerMap::value_type(rname, nullptr));
296 
297  if (inserted.second) {

299  inserted.first->second = pRequestHandler;
