96  void HttpRunControl::receiveStartOfRunPost(Mongoose::Request &request, Mongoose::StreamResponse &response) {
97  dqm_info("Received POST: /{0}/SOR", this->runControl().name() );
98 
99  if(runControl().isRunning()) {
100  response << "ERROR: Couldn't start new run, already running !" << std::endl;
101  return;
102  }
103 
104  std::map<std::string, std::string> variables(request.getAllVariable());
105  std::map<std::string, std::string> urlVariables;
106  request.getVars(urlVariables);
107 
108  dqm4hep::core::Run newRun;
109  std::string password;
110  bool consistent = false;
111 
112  for(auto iter = variables.begin(), endIter = variables.end() ; endIter != iter ; ++iter) {
113  if(iter->first == "run") {
114  int runNumber = 0;
115  dqm4hep::core::stringToType(iter->second, runNumber);
116  newRun.setRunNumber(runNumber);
117  consistent = true;
118  continue;
119  }
120  if(iter->first == "description") {
121  newRun.setDescription(iter->second);
122  continue;
123  }
124  if(iter->first == "detectorName") {
125  newRun.setDetectorName(iter->second);
126  continue;
127  }
128  if(iter->first == "password") {
129  password = iter->second;
130  continue;
131  }
132  newRun.setParameter(iter->first, iter->second);
133  }
134  if(not consistent) {
135  response << "ERROR: Run number required to start a new run !!"<< std::endl;
136  return;
137  }
138  try {
139  this->startNewRun(newRun, password);
140  dqm_info( "*** Started new run ***" );
141  dqm_info( "{0}", core::typeToString(newRun) );
142  dqm_info( "***********************" );
143  }
144  catch( dqm4hep::core::StatusCodeException &exception ) {
145  response << "ERROR: Couldn't start new run ! What :" << exception.toString()<< std::endl;
146  return;
147  }
148  response << "SUCCESS\n";
149  }

153  void HttpRunControl::receiveEndOfRunPost(Mongoose::Request &request, Mongoose::StreamResponse &response) {
154  dqm_info("Received POST: /{0}/EOR", this->runControl().name());
155  if(! this->runControl().isRunning()) {
156  response << "ERROR: Not running ! Couldn't end current run !";
157  return;
158  }
159  std::map<std::string, std::string> variables(request.getAllVariable());
160  std::string password;
161  dqm4hep::core::StringMap parameters;
162 
163  for(auto iter = variables.begin(), endIter = variables.end() ; endIter != iter ; ++iter) {
164  if( iter->first == "password" ) {
165  password = iter->second;
166  continue;
167  }
168  parameters[iter->first] = iter->second;
169  }
170 
171  try {
172  this->endCurrentRun(parameters, password);
173  }
174  catch( dqm4hep::core::StatusCodeException &exception ) {
175  response << "ERROR: Couldn't stop current run ! Exception: " << exception.toString() << std::endl;
176  return;
177  }
178 
179  response << "SUCCESS";
180  }
