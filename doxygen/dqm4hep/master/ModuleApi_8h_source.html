258  template <typename ObjectType, typename... Args>
259  inline core::StatusCode ModuleApi::bookObject(const Module *const module, OnlineElementPtr &monitorElement, const std::string &path, const std::string &name, const std::string &title, Args... args) {
260  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, ModuleApi::checkModule(module));
261  if(not module->moduleApplication()->allowBooking()) {
262  dqm_error( "Module application allows monitor element booking only in readSettings() or init() functions !" );
263  return core::STATUS_CODE_NOT_ALLOWED;
264  }
265  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, module->moduleApplication()->monitorElementManager()->bookObject<ObjectType>(path, name, title, monitorElement, args...));
266  monitorElement->setModuleName(module->name());
267  return core::STATUS_CODE_SUCCESS;
268  }
