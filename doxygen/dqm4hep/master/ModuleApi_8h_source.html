277  inline core::StatusCode ModuleApi::bookObject(const Module *const module, OnlineElementPtr &monitorElement, const std::string &path, const std::string &name, const std::string &title, Args... args) {
278  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, ModuleApi::checkModule(module));
279  if(not module->moduleApplication()->allowBooking()) {
280  dqm_error( "Module application allows monitor element booking only in readSettings() or init() functions !" );
281  return core::STATUS_CODE_NOT_ALLOWED;
282  }
283  RETURN_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, module->moduleApplication()->monitorElementManager()->bookObject<ObjectType>(path, name, title, monitorElement, args...));
284  monitorElement->setModuleName(module->name());
285  return core::STATUS_CODE_SUCCESS;
286  }
