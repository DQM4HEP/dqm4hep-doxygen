67  void EventCollectorClient::setUpdateMode(const std::string &source, bool receiveUpdates) {
68  auto findIter = m_sourceInfoMap.find(source);
69  if(m_sourceInfoMap.end() == findIter) {
70  dqm_error( "EventCollectorClient::setUpdateMode: no source with '{0}' registered !", source );
71  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
72  }
73  bool subscribed = m_client.hasSubscribed(OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source));
74 
75  // two cases:
76  // 1) already subscribed and don't want to receive updates anymore
77  // 2) not subscribed and want to receive updates
78  if(subscribed != receiveUpdates) {
79  if(receiveUpdates) {
80  m_client.subscribe(
81  OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source),
82  &findIter->second,
83  &EventCollectorClient::SourceInfo::receiveEvent
84  );
85  }
86  else {
87  m_client.unsubscribe(
88  OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source),
89  &findIter->second
90  );
91  }
92  }
93  }

127  core::EventPtr EventCollectorClient::readEvent(const net::Buffer &buffer) {
128  std::lock_guard<std::recursive_mutex> lock(m_mutex);
129 
130  try {
131  m_buffer.SetBuffer((void*)buffer.begin(), buffer.size(), false);
132  }
133  catch(...) {
134  dqm_error( "EventCollectorClient::readEvent: couldn't setup buffer device !" );
135  return nullptr;
136  }
137  // read event using event streamer
138  core::EventPtr event(nullptr);
139  core::StatusCode statusCode = m_eventStreamer.readEvent(event, m_buffer);
140 
141  if(core::STATUS_CODE_SUCCESS != statusCode) {
142  dqm_error( "EventCollectorClient::readEvent: streamer couldn't read event: {0}", core::statusCodeToString(statusCode) );
143  return nullptr;
144  }
145  return event;
146  }
