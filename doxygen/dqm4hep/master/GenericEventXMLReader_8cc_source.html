54  core::StatusCode GenericEventXMLReader::open(const std::string &fname) {
55  if(not m_document.LoadFile(fname)) {
56  dqm_error("GenericEventXMLReader::open(): couldn't load xml file ['{0}', row: {1}, col: {2}] : {3}", fname,
57  m_document.ErrorRow(), m_document.ErrorCol(), m_document.ErrorDesc());
58  return STATUS_CODE_FAILURE;
59  }
60  m_currentEvent = m_document.RootElement()->FirstChildElement("event");
61  if(nullptr == m_currentEvent) {
62  dqm_error( "No event stored in the file '{0}'", fname );
63  return STATUS_CODE_FAILURE;
64  }
65  return STATUS_CODE_SUCCESS;
66  }

70  core::StatusCode GenericEventXMLReader::skipNEvents(int nEvents) {
71  int nSkippedEvents = 0;
72  while(1) {
73  if(nSkippedEvents == nEvents)
74  break;
75  m_currentEvent = m_currentEvent->NextSiblingElement("event");
76  if(nullptr == m_currentEvent) {
77  dqm_error( "Couldn't skip {0} events in file '{1}' : file has less than {1}", nEvents, m_document.ValueStr() );
78  return STATUS_CODE_OUT_OF_RANGE;
79  }
80  nSkippedEvents++;
81  }
82  return STATUS_CODE_SUCCESS;
83  }

87  core::StatusCode GenericEventXMLReader::runInfo(core::Run &run) {
88  auto runInfoElement = m_document.RootElement()->FirstChildElement("run");
89  if(!runInfoElement) {
90  dqm_error( "No run info available in file '{0}'", m_document.ValueStr() );
91  return STATUS_CODE_NOT_FOUND;
92  }
93  TiXmlHandle handle(runInfoElement);
94  int runNumber = 0;
95  int32_t startTime = 0, endTime = 0;
96  std::string description, detectorName;
97  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
98  "RunNumber", runNumber));
99  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
100  "StartTime", startTime));
101  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
102  "EndTime", endTime));
103  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
104  "Description", description));
105  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
106  "DetectorName", detectorName));
107  auto parameters = runInfoElement->FirstChildElement("parameters");
108  if(nullptr != parameters) {
109  for(auto child = parameters->FirstChildElement("parameter") ; nullptr != child ; child = child->NextSiblingElement("parameter")) {
110  std::string name, value;
111  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::parseParameterElement(child, name, value));
112  run.setParameter(name, value);
113  }
114  }
115  run.setRunNumber(runNumber);
116  run.setStartTime(core::time::asPoint(startTime));
117  run.setEndTime(core::time::asPoint(endTime));
118  run.setDescription(description);
119  run.setDetectorName(detectorName);
120  return STATUS_CODE_SUCCESS;
121  }

125  core::StatusCode GenericEventXMLReader::readNextEvent() {
126  EventPtr event = GenericEvent::make_shared();
127  GenericEvent *generic = event->getEvent<GenericEvent>();
128  // end of file !
129  if(nullptr == m_currentEvent) {
130  return STATUS_CODE_OUT_OF_RANGE;
131  }
132  TiXmlHandle handle(m_currentEvent);
133  int32_t eventType = static_cast<int32_t>(UNKNOWN_EVENT);
134  std::string source;
135  int32_t timeStamp = 0, eventNumber = 0, runNumber = 0;
136  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
137  "EventType", eventType));
138  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
139  "Source", source));
140  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
141  "TimeStamp", timeStamp));
142  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
143  "EventNumber", eventNumber));
144  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
145  "RunNumber", runNumber));
146  event->setType(static_cast<EventType>(eventType));
147  event->setSource(source);
148  event->setTimeStamp(core::time::asPoint(timeStamp));
149  event->setEventNumber(eventNumber);
150  event->setRunNumber(runNumber);
151  // loop over <field> elements
152  for(auto field = m_currentEvent->FirstChildElement("field") ; nullptr != field ; field = field->NextSiblingElement("field")) {
153  std::string name, type;
154  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(field, "name", name));
155  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(field, "type", type));
156  if("int" == type) {
157  if(nullptr == field->GetText()) {
158  return STATUS_CODE_FAILURE;
159  }
160  std::string content = field->GetText();
161  IntVector values;
162  core::tokenize(content, values, " ");
163  generic->setValues(name, values);
164  }
165  else if("float" == type) {
166  if(nullptr == field->GetText()) {
167  return STATUS_CODE_FAILURE;
168  }
169  std::string content = field->GetText();
170  FloatVector values;
171  core::tokenize(content, values, " ");
172  generic->setValues(name, values);
173  }
174  else if("double" == type) {
175  if(nullptr == field->GetText()) {
176  return STATUS_CODE_FAILURE;
177  }
178  std::string content = field->GetText();
179  DoubleVector values;
180  core::tokenize(content, values, " ");
181  generic->setValues(name, values);
182  }
183  else if("string" == type) {
184  StringVector values;
185  // Wrap string into <str> </str> elements to avoid string tokenize
186  for(auto strElement = field->FirstChildElement("str") ; nullptr != strElement ; strElement = strElement->NextSiblingElement("str")) {
187  values.push_back(strElement->GetText() != nullptr ? strElement->GetText() : "");
188  }
189  generic->setValues(name, values);
190  }
191  else {
192  dqm_warning( "Unrecognized field type '{0}' ! Skipping ...", type );
193  continue;
194  }
195  }
196  onEventRead().emit(event);
197  m_currentEvent = m_currentEvent->NextSiblingElement("event");
198  return STATUS_CODE_SUCCESS;
199  }
