224  inline StatusCode DBInterface::query(const std::string &uquery, T &result) {
225  if (!this->isConnected())
226  return STATUS_CODE_NOT_INITIALIZED;
227 
228  if (mysql_query(m_pMySQL, uquery.c_str())) {
229  dqm_error("MySQL query failed : {0}", mysql_error(m_pMySQL));
230  return STATUS_CODE_FAILURE;
231  }
232 
233  MYSQL_RES *pMySQLResult = mysql_store_result(m_pMySQL);
234 
235  if (!pMySQLResult) {
236  dqm_error("MySQL store result failed : {0}", mysql_error(m_pMySQL));
237  return STATUS_CODE_FAILURE;
238  }
239 
240  MYSQL_ROW row = mysql_fetch_row(pMySQLResult);
241 
242  if (!dqm4hep::core::stringToType(row[0], result)) {
243  mysql_free_result(pMySQLResult);
244  return STATUS_CODE_FAILURE;
245  }
246 
247  mysql_free_result(pMySQLResult);
248 
249  return STATUS_CODE_SUCCESS;
250  }
