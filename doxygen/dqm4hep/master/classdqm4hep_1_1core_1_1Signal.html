Detailed Description

template<typename... Args>
class dqm4hep::core::Signal< Args >

Signal class.

Implement the observer pattern a la Qt. Example:

void func() {
std::cout << "Hello world !" << std::endl;
}
class Toto {
public:
void hello() {
std::cout << "Toto world !" << std::endl;
}
};
Signal<> sig;
Toto toto;
sig.connect(func);
sig.connect(&toto, &Toto::hello);
sig.emit();

Allows nice and understandable piece of code like:

// Holds a property and signal
class Property {
public:
void setProperty(int value) {
m_property = value;
m_onUpdate.emit(m_property);
}
Signal<int> &onUpdate() {
return m_onUpdate;
}
private:
int m_property = {0};
Signal<int> m_onUpdate = {};
};
// To print the property !
class Printer {
public:
void print(int value) {
std::cout << "Value is " << value << std::endl;
}
};
Printer printer;
Property property;
property.onUpdate().connect(&printer, &Printer::print); // I love this line !
property.setProperty(42); // Calls 'print' method
