208  void WsServer::start() {
209  if(m_running.load()) {
210  return;
211  }
212  // Set logging settings
213  m_server.set_access_channels(websocketpp::log::alevel::none);
214 
215  // Initialize Asio
216  m_server.init_asio();
217 
218  // Register our message handler
219  m_server.set_message_handler(std::bind(&WsServer::onMessage, this, _1, _2));
220  m_server.set_open_handler(std::bind(&WsServer::onOpen, this, _1));
221  m_server.set_close_handler(std::bind(&WsServer::onClose, this, _1));
222 
223  // Start listening
224  m_server.listen(m_port);
225 
226  // Start the server accept loop
227  m_server.start_accept();
228 
229  // Start the ASIO io_service run loop
230  m_thread = std::thread(&server::run, std::ref(m_server));
231  m_running = true;
232  }

254  void WsServer::onClose(connection_hdl hdl) {
255  std::lock_guard<std::recursive_mutex> lock(m_mutex);
256  server::connection_ptr con = m_server.get_con_from_hdl(hdl);
257  const std::string serviceName = con->get_resource();
258  auto findIter = m_serviceMap.find(serviceName);
259  if(m_serviceMap.end() == findIter) {
260  return;
261  }
262  // remove service subscriber (if subscribed)
263  if(findIter->second->type() == SERVICE_TYPE) {
264  auto findIter2 = m_serviceConnections.find(serviceName);
265  if(findIter2 == m_serviceConnections.end()) {
266  return;
267  }
268  findIter2->second.erase(con);
269  }
270  }

274  void WsServer::onMessage(connection_hdl hdl, message_ptr msg) {
275  std::lock_guard<std::recursive_mutex> lock(m_mutex);
276  server::connection_ptr con = m_server.get_con_from_hdl(hdl);
277  const std::string serviceName = con->get_resource();
278  auto findIter = m_serviceMap.find(serviceName);
279 
280  if(findIter == m_serviceMap.end()) {
281  m_server.close(hdl, websocketpp::close::status::normal, "Service '" + serviceName + "' not available !");
282  return;
283  }
284  // handle service subscription / unsubscription
285  if(findIter->second->type() == SERVICE_TYPE) {
286  onServiceMessage(serviceName, hdl, msg);
287  }
288  else if(findIter->second->type() == COMMAND_TYPE) {
289  onCommandMessage(serviceName, hdl, msg);
290  }
291  else if(findIter->second->type() == RPC_TYPE) {
292  onRpcMessage(serviceName, hdl, msg);
293  }
294  else {
295  m_server.close(hdl, websocketpp::close::status::normal, "Service '" + serviceName + "' unknown type !");
296  return;
297  }
298  }

303  std::lock_guard<std::recursive_mutex> lock(m_mutex);

316  void WsServer::onServiceMessage(const std::string &serviceName, connection_hdl hdl, message_ptr msg) {
317  server::connection_ptr con = m_server.get_con_from_hdl(hdl);
318  auto findIter2 = m_serviceConnections.find(serviceName);
319  // handle subscription/un-subscription to/from service
320  if(msg->get_payload() == "subscribe") {
321  if(findIter2 == m_serviceConnections.end()) {
322  m_server.close(hdl, websocketpp::close::status::normal, "Internal error: service '" + serviceName + "' not available !");
323  return;
324  }
325  // insert new subscriber
326  findIter2->second.insert(con);
327  m_server.send(hdl, "ok", websocketpp::frame::opcode::text);
328  return;
329  }
330  else if(msg->get_payload() == "unsubscribe") {
331  if(findIter2 == m_serviceConnections.end()) {
332  m_server.close(hdl, websocketpp::close::status::normal, "Internal error: service '" + serviceName + "' not available !");
333  return;
334  }
335  // remove subscriber
336  findIter2->second.erase(con);
337  m_server.send(hdl, "ok", websocketpp::frame::opcode::text);
338  return;
339  }
340  else {
341  m_server.close(hdl, websocketpp::close::status::normal, "Unknown command for service '" + serviceName + "', command: " + msg->get_payload() + " !");
342  return;
343  }
344  }
