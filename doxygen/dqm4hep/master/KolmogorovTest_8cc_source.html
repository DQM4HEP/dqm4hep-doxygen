91  void KolmogorovTest::userRun(MonitorElement* pMonitorElement, QualityTestReport &report) {
92  const bool hasObject = (pMonitorElement->object() != nullptr);
93  const bool hasReference = (pMonitorElement->reference() != nullptr);
94 
95  if (!hasObject) {
96  report.m_message = "No object attached to monitor element";
97  report.m_quality = 0.f;
98  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
99  }
100 
101  if (!hasReference) {
102  report.m_message = "No reference attached to monitor element";
103  report.m_quality = 0.f;
104  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
105  }
106 
107  const bool isObjHistogram = pMonitorElement->object()->InheritsFrom("TH1");
108  const bool isObjGraph = pMonitorElement->object()->InheritsFrom("TGraph");
109  const bool isRefHistogram = pMonitorElement->reference()->InheritsFrom("TH1");
110  const bool isRefGraph = pMonitorElement->reference()->InheritsFrom("TGraph");
111 
112  if ( (isObjGraph and !isRefGraph) or (isObjHistogram and !isRefHistogram) ) {
113  report.m_message = "The object and reference must be of the same type";
114  report.m_quality = 0.f;
115  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
116  }
117 
118  if (isObjGraph) {
119  TGraph* pGraph = pMonitorElement->objectTo<TGraph>();
120  TGraph* pReferenceGraph = pMonitorElement->referenceTo<TGraph>();
121  std::string options = KolmogorovTest::getTestOptions(isObjHistogram);
122 
123  int sizeGraph = pGraph->GetN();
124  int sizeRef = pReferenceGraph->GetN();
125 
126  if ( (sizeGraph == 0) or (sizeRef == 0) ) {
127  report.m_message = "One or more of the graphs are empty!";
128  report.m_quality = 0.0;
129  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
130  }
131 
132  pGraph->Sort();
133  pReferenceGraph->Sort();
134  double* pArrayGraph = pGraph->GetY();
135  double* pArrayRef = pReferenceGraph->GetY();
136 
137  report.m_extraInfos["options"] = options;
138  report.m_quality = TMath::KolmogorovTest(sizeGraph, pArrayGraph, sizeRef, pArrayRef, options.c_str());
139  }
140  else if (isObjHistogram) {
141  TH1* pHistogram = pMonitorElement->objectTo<TH1>();
142  TH1* pReferenceHistogram = pMonitorElement->referenceTo<TH1>();
143  std::string options = KolmogorovTest::getTestOptions(isObjHistogram);
144 
145  report.m_extraInfos["options"] = options;
146  report.m_message = "The Kolmogorov test isn't intended for histograms! Use caution when interepreting the quality.";
147  report.m_quality = pHistogram->KolmogorovTest(pReferenceHistogram, options.c_str());
148  }
149  else {
150  report.m_message = "ROOT monitor element object is of unrecognised type!";
151  report.m_quality = 0.f;
152  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
153  }
154 
155  }
