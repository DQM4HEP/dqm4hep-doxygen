40  StatusCode Archiver::open(const std::string &fname, const std::string &opMode, bool overwrite, int runNumber) {
41  // if already open write the archive if not done
42  // and close it before to re-open
43  if (isOpened()) {
44  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, close());
45  }
46  if (fname.empty()) {
47  return STATUS_CODE_INVALID_PARAMETER;
48  }
49  int fileId(0);
50  size_t pos = fname.rfind(".root");
51  if (std::string::npos == pos) {
52  dqm_error("Couldn't open archive '{0}' ! Must be a root file !", fname);
53  return STATUS_CODE_INVALID_PARAMETER;
54  }
55  std::string baseArchiveName = fname.substr(0, pos);
56  std::string fullArchiveName = fname;
57  if (not overwrite) {
58  while (!gSystem->AccessPathName(fullArchiveName.c_str())) {
59  std::stringstream ss;
60  ss << baseArchiveName;
61  if(runNumber >= 0) {
62  ss << "_I" << runNumber;
63  }
64  ss << "_" << fileId << ".root";
65  fullArchiveName = ss.str();
66  fileId++;
67  }
68  m_fileName = fullArchiveName;
69  }
70  else if(runNumber >= 0) {
71  std::stringstream ss;
72  ss << baseArchiveName << "_I" << runNumber << ".root";
73  m_fileName = ss.str();
74  }
75  else {
76  m_fileName = fname;
77  }
78  m_openingMode = opMode;
79  dqm_info("Archiver::open: Opening archive {0}", m_fileName);
80  m_file.reset(new TFile(m_fileName.c_str(), m_openingMode.c_str()));
81  if (nullptr == m_file) {
82  dqm_error("Archiver::open: Couldn't open archive '{0}' !", m_fileName);
83  return STATUS_CODE_FAILURE;
84  }
85  m_isOpened = true;
86  return STATUS_CODE_SUCCESS;
87  }

169  StatusCode Archiver::recursiveWrite(MonitorElementDir directory, TDirectory *rootDirectory, const std::string &refSuffix) {
170  if (nullptr == directory || nullptr == rootDirectory) {
171  return STATUS_CODE_INVALID_PTR;
172  }
173  rootDirectory->cd();
174  const auto &subDirList(directory->subdirs());
175  for (auto iter = subDirList.begin(), endIter = subDirList.end(); endIter != iter; ++iter) {
176  auto subDir = *iter;
177  TDirectory *rootSubDirectory = rootDirectory->mkdir(subDir->name().c_str());
178  if (nullptr != rootSubDirectory) {
179  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, Archiver::recursiveWrite(subDir, rootSubDirectory, refSuffix));
180  }
181  }
182  // write the monitor elements
183  if (not directory->isEmpty()) {
184  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, Archiver::writeMonitorElements(directory, rootDirectory, refSuffix));
185  }
186  return STATUS_CODE_SUCCESS;
187  }

191  StatusCode Archiver::writeMonitorElements(MonitorElementDir directory, TDirectory *rootDirectory, const std::string &refSuffix) {
192  if (nullptr == directory || nullptr == rootDirectory) {
193  return STATUS_CODE_INVALID_PTR;
194  }
195  rootDirectory->cd();
196  const auto &monitorElementList(directory->contents());
197  for (auto iter = monitorElementList.begin(), endIter = monitorElementList.end(); endIter != iter; ++iter) {
198  if(not m_selectorFunction(*iter)) {
199  continue;
200  }
201  const std::string writeName = (*iter)->name();
202  TObject *object = (*iter)->object();
203  if(nullptr != object) {
204  rootDirectory->WriteObjectAny(object, object->IsA(), writeName.c_str());
205  TObject *reference = (*iter)->reference();
206  if(not refSuffix.empty() and nullptr != reference) {
207  rootDirectory->WriteObjectAny(reference, reference->IsA(), (writeName + refSuffix).c_str());
208  }
209  }
210  }
211  return STATUS_CODE_SUCCESS;
212  }

217  ArchiverSelector::ArchiverSelector() {
218  m_function = [this](MonitorElementPtr element)->bool{
219  if(m_selectorFunctions.empty()) {
220  return true;
221  }
222  for(auto &selector : m_selectorFunctions) {
223  if(selector(element)) {
224  dqm_debug( "Archiving element path: {0}, name: {1} ...", element->path(), element->name() );
225  return true;
226  }
227  }
228  dqm_debug( "Skipping element path: {0}, name: {1} !", element->path(), element->name() );
229  return false;
230  };
