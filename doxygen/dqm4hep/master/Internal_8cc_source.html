31  bool wildcardMatch(const std::string &testString, const std::string &wildcardString, bool caseSensitive) {
32  std::string wildcardPattern = wildcardString;
33  replaceAll(wildcardPattern, "\\", "\\\\");
34  replaceAll(wildcardPattern, "^", "\\^");
35  replaceAll(wildcardPattern, ".", "\\.");
36  replaceAll(wildcardPattern, "$", "\\$");
37  replaceAll(wildcardPattern, "|", "\\|");
38  replaceAll(wildcardPattern, "(", "\\(");
39  replaceAll(wildcardPattern, ")", "\\)");
40  replaceAll(wildcardPattern, "[", "\\[");
41  replaceAll(wildcardPattern, "]", "\\]");
42  replaceAll(wildcardPattern, "*", "\\*");
43  replaceAll(wildcardPattern, "+", "\\+");
44  replaceAll(wildcardPattern, "?", "\\?");
45  replaceAll(wildcardPattern, "/", "\\/");
46  // Convert chars '*?' back to their regex equivalents
47  replaceAll(wildcardPattern, "\\?", ".");
48  replaceAll(wildcardPattern, "\\*", ".*");
49  // regex !
50  auto flag = caseSensitive ? std::regex::ECMAScript : std::regex::ECMAScript | std::regex::icase;
51  std::regex pattern(wildcardPattern, flag);
52  return std::regex_match(testString, pattern);
53  }

96  } else {
97 
98  double prevCpuTimeUser = stats.cpuTimeUser;
99  double prevCpuTimeSys = stats.cpuTimeSys;
100  stats.cpuTimeUser = ru.ru_utime.tv_sec + ru.ru_utime.tv_usec / 1000000.;
101  stats.cpuTimeSys = ru.ru_stime.tv_sec + ru.ru_stime.tv_usec / 1000000.;
102 
103  struct timeval timeNow;
104  gettimeofday(&timeNow, NULL);
105  double start = stats.lastPollTime.tv_sec + stats.lastPollTime.tv_usec / 1000000.;
106  double stop = timeNow.tv_sec + timeNow.tv_usec / 1000000.;
107  stats.lastPollTime = timeNow;
108  if (stop > start) {
109  double timeDif = stop - start;
110  // Don't compute percentage for the first call
111  if (0 != prevCpuTimeUser || 0 != prevCpuTimeSys) {
112  stats.cpuUser = (stats.cpuTimeUser - prevCpuTimeUser) / timeDif;
113  stats.cpuSys = (stats.cpuTimeSys - prevCpuTimeSys) / timeDif;
114  stats.cpuTot = stats.cpuUser + stats.cpuSys;
115  stats.cpuUser *= 100.;
116  stats.cpuSys *= 100.;
117  stats.cpuTot *= 100.;
118  }
119  } else {
120  dqm_error("[{0}] - Something went wrong when comparing clocks", __FUNCTION__);
121  stats.cpuUser = -1;
122  stats.cpuSys = -1;
123  stats.cpuTot = -1;
124  }
125  }

549  static void darwinProcStats(ProcessStats &stats) {
550 #ifdef _LP64
551 #define vm_region vm_region_64
552 #endif
553 
554 // taken from <mach/shared_memory_server.h> which is obsoleted in 10.5
555 #define GLOBAL_SHARED_TEXT_SEGMENT 0x90000000U
556 #define GLOBAL_SHARED_DATA_SEGMENT 0xA0000000U
557 #define SHARED_TEXT_REGION_SIZE 0x10000000
558 #define SHARED_DATA_REGION_SIZE 0x10000000
559 
560  procCpuStats(stats);
561 
562  task_t a_task = mach_task_self();
563  task_basic_info_data_t ti;
564  mach_msg_type_number_t count = TASK_BASIC_INFO_COUNT;
565 
566  kern_return_t kr = task_info(a_task, TASK_BASIC_INFO, (task_info_t)&ti, &count);
567  if (kr != KERN_SUCCESS) {
568  dqm_error("[{0}] - Failed to get task_info: {1}", __FUNCTION__, mach_error_string(kr));
569  throw core::StatusCodeException(STATUS_CODE_FAILURE);
570  } else {
571  // resident size does not require any calculation. Virtual size
572  // needs to be adjusted if traversing memory objects do not include the
573  // globally shared text and data regions
574  mach_port_t object_name;
575  vm_address_t address;
576  vm_region_top_info_data_t info;
577 
578  vm_size_t rsize = ti.resident_size;
579  vm_size_t vsize = ti.virtual_size;
580  vm_size_t vprvt = 0;
581  vm_size_t size = 0;
582 
583  for (address = 0;; address += size) {
584  // get memory region
585  count = VM_REGION_TOP_INFO_COUNT;
586  if (vm_region(a_task, &address, &size, VM_REGION_TOP_INFO, (vm_region_info_t)&info, &count, &object_name) !=
587  KERN_SUCCESS) {
588  // no more memory regions.
589  break;
590  }
591 
592  if (address >= GLOBAL_SHARED_TEXT_SEGMENT &&
593  address < (GLOBAL_SHARED_DATA_SEGMENT + SHARED_DATA_REGION_SIZE)) {
594  // This region is private shared.
595  // Check if this process has the globally shared
596  // text and data regions mapped in. If so, adjust
597  // virtual memory size and exit loop.
598  if (info.share_mode == SM_EMPTY) {
599  vm_region_basic_info_data_64_t b_info;
600  count = VM_REGION_BASIC_INFO_COUNT_64;
601  if (vm_region_64(a_task, &address, &size, VM_REGION_BASIC_INFO, (vm_region_info_t)&b_info, &count,
602  &object_name) != KERN_SUCCESS) {
603  break;
604  }
605 
606  if (b_info.reserved) {
607  vsize -= (SHARED_TEXT_REGION_SIZE + SHARED_DATA_REGION_SIZE);
608  // break; // only for vsize
609  }
610  }
611  // Short circuit the loop if this isn't a shared
612  // private region, since that's the only region
613  // type we care about within the current address range.
614  if (info.share_mode != SM_PRIVATE) {
615  continue;
616  }
617  }
618  switch (info.share_mode) {
619  case SM_COW: {
620  if (info.ref_count == 1) {
621  vprvt += size;
622  } else {
623  vprvt += info.private_pages_resident * getpagesize();
624  }
625  break;
626  }
627  case SM_PRIVATE: {
628  vprvt += size;
629  break;
630  }
631  default:
632  break;
633  }
634  }
635 
636  stats.vm = (long)(rsize / 1024.);
637  stats.rss = (long)(vprvt / 1024.);
638  }
639  }

664  sysctlBuffer = (uint8_t *)malloc(currentSize);

668  throw core::StatusCodeException(STATUS_CODE_FAILURE);
669  }
670  sysctlBufferSize = currentSize;
671  }
672 
673  // Read in new data
674  if (sysctl(mib, 6, sysctlBuffer, &currentSize, NULL, 0) != 0) {
675  const int errNum = errno;
676  dqm_error("[{0}] - Failed to read network statistics: {1}", __FUNCTION__, errNum);
677  throw core::StatusCodeException(STATUS_CODE_FAILURE);
678  }
679 
680  // Walk through the reply
681  uint8_t *currentData = sysctlBuffer;
682  uint8_t *currentDataEnd = sysctlBuffer + currentSize;
683 
684  while (currentData < currentDataEnd) {
685  // Expecting interface data

687  if (ifmsg->ifm_type != RTM_IFINFO) {
688  currentData += ifmsg->ifm_msglen;
689  continue;
690  }
691 
692  // Only look at link layer items
693  struct sockaddr_dl *sdl = (struct sockaddr_dl *)(ifmsg + 1);
694  if (sdl->sdl_family != AF_LINK) {
695  currentData += ifmsg->ifm_msglen;
696  continue;
697  }
698

700  char ifc_name[32];

702  ifc_name[sdl->sdl_nlen] = 0;
703 
704  INetworkStats stat;
705  stat.tot_rcv_kbytes = ifmsg->ifm_data.ifi_ibytes / unit;
706  stat.tot_rcv_packets = ifmsg->ifm_data.ifi_ipackets;
707  stat.tot_rcv_errs = ifmsg->ifm_data.ifi_ierrors;
708  stat.tot_snd_kbytes = ifmsg->ifm_data.ifi_obytes / unit;
709  stat.tot_snd_packets = ifmsg->ifm_data.ifi_opackets;
710  stat.tot_snd_errs = ifmsg->ifm_data.ifi_oerrors;
711  stats[ifc_name] = stat;
712

714  }
