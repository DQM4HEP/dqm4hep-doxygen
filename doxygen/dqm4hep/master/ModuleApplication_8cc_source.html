136  void ModuleApplication::onEvent(AppEvent * appEvent) {
137  if(AppEvent::SERVICE_UPDATE == appEvent->type()) {
138  ServiceUpdateEvent* svc = dynamic_cast<ServiceUpdateEvent*>(appEvent);
139  // Start of run
140  if(svc->serviceName() == OnlineRoutes::RunControl::sor(m_runControl.name())) {
141  processStartOfRun(svc);
142  }
143  // End of run
144  if(svc->serviceName() == OnlineRoutes::RunControl::eor(m_runControl.name())) {
145  processEndOfRun();
146  }
147  }
148  if(AppEvent::COMMAND_HANDLING == appEvent->type()) {
149  CommandEvent *cmd = dynamic_cast<CommandEvent*>(appEvent);
150  // update subscription list
151  if(cmd->commandName() == OnlineRoutes::ModuleApplication::subscribe(name())) {
152  receiveSubscriptionList(cmd);
153  }
154  }
155  if(AppEvent::END_OF_RUN == appEvent->type()) {
156  auto eorEvent = dynamic_cast<StoreEvent<core::Run>*>(appEvent);
157  auto run = eorEvent->data();
158  m_runControl.endCurrentRun(run.parameters());
159  if(EVENT_READER == appRunningMode()) {
160  dqm_info( "End of run processed. Exiting application ..." );
161  this->exit(0);
162  }
163  }
164  // process event received from the event collector
165  if(AppEvent::PROCESS_EVENT == appEvent->type() and ANALYSIS == appModuleType()) {
166  if(m_currentNQueuedEvents != 0) {
167  m_currentNQueuedEvents--;
168  }
169  if(m_runControl.isRunning()) {
170  auto procEvent = dynamic_cast<StoreEvent<core::EventPtr>*>(appEvent);
171  auto anaModule = moduleAs<AnalysisModule>();
172  anaModule->process(procEvent->data());
173  m_cycle.incrementCounter();
174  if(EVENT_READER == appRunningMode()) {
175  auto status = m_eventReader->readNextEvent();
176  // end of file ?
177  if(status == core::STATUS_CODE_OUT_OF_RANGE) {
178  processEndOfRun();
179  }
180  else if(status != core::STATUS_CODE_SUCCESS) {
181  dqm_error( "Error while reading event: file reader returned status '{0}'", core::statusCodeToString(status) );
182  this->exit(1);
183  }
184  }
185  }
186  }
187  // generic loop
188  if(AppEvent::PROCESS_EVENT == appEvent->type() and STANDALONE == appModuleType()) {
189  auto standModule = moduleAs<StandaloneModule>();
190  standModule->process();
191  m_cycle.incrementCounter();
192  m_standaloneTimer->start();
193  }
194  // end of cycle
195  if(AppEvent::END_OF_CYCLE == appEvent->type()) {
196  auto eocEvent = dynamic_cast<StoreEvent<EOCCondition>*>(appEvent);
197  auto condition = eocEvent->data();
198  m_module->endOfCycle(condition);
199  if(condition.m_counter > 0) {
200  try {
201  core::QReportStorage reportStorage;
202  // process quality tests
203  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_monitorElementManager->runQualityTests(reportStorage));
204  core::json jreports;
205  reportStorage.toJson(jreports);
206  dqm_info( jreports.dump(2) );
207 
208  OnlineElementPtrList publishElements;
209  m_monitorElementManager->iterate<OnlineElement>([&](OnlineElementPtr monitorElement){
210  if(not monitorElement->publish() or not monitorElement->subscribed()){
211  return true;
212  }
213  publishElements.push_back(monitorElement);
214  return true;
215  });
216  // TODO send monitor element to collector
217  }
218  catch(core::StatusCodeException &exception) {
219  dqm_error( "Error caught at end of cycle: {0}", exception.getStatusCode() );
220  }
221  }
222  // always restart a new cycle for standalone modules
223  if(STANDALONE == appModuleType()) {
224  m_module->startOfCycle();
225  m_cycle.startCycle(true);
226  }
227  // forced end is a synonym of end of run for analysis modules
228  // do not restart a cycle in this case
229  if(ANALYSIS == appModuleType() and not condition.m_forcedEnd) {
230  m_module->startOfCycle();
231  m_cycle.startCycle(true);
232  }
233  }
234  }

238  void ModuleApplication::onStart() {
239  if(ONLINE == appRunningMode()) {
240  // get run control status in case it is already running
241  sendRequest(OnlineRoutes::RunControl::status(m_runControl.name()), net::Buffer(), [this](const net::Buffer &response){
242  core::json statusJson = core::json::parse(response.begin(), response.end());
243  bool rcRunning = statusJson.value<bool>("running", false);
244  if(rcRunning) {
245  core::json runJson = statusJson.value<core::json>("run", core::json({}));
246  core::Run run;
247  run.fromJson(runJson);
248  m_runControl.startNewRun(run);
249  }
250  });
251  }
252  if(STANDALONE == appModuleType()) {
253  m_module->startOfCycle();
254  m_cycle.startCycle(true);
255  m_standaloneTimer->start();
256  }
257  else if(ANALYSIS == appModuleType() and m_runControl.isRunning()) {
258  m_module->startOfCycle();
259  m_cycle.startCycle(true);
260  }
261  if(EVENT_READER == appRunningMode()) {
262  core::Run run;
263  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_eventReader->runInfo(run));
264  m_runControl.startNewRun(run);
265  m_module->startOfCycle();
266  m_cycle.startCycle();
267  // read first will post an event in event loop
268  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_eventReader->readNextEvent());
269  }
270  }

330  void ModuleApplication::parseSteeringFile(const std::string &fname) {
331  m_parser.parse(fname);
332 
333  auto document = m_parser.document();
334  const core::TiXmlHandle xmlHandle(document.RootElement());
335  auto storageElement = xmlHandle.FirstChildElement("storage").Element();
336  auto settingsElement = xmlHandle.FirstChildElement("settings").Element();
337  auto moduleElement = xmlHandle.FirstChildElement("module").Element();
338  auto archiverElement = xmlHandle.FirstChildElement("archiver").Element();
339 
340  if(nullptr != storageElement) {
341  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_monitorElementManager->parseStorage<OnlineElement>(storageElement));
342  }
343 
344  if(nullptr == settingsElement) {
345  dqm_error("parseSteeringFile: Missing <settings> section !");
346  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
347  }
348 
349  // determine running mode
350  static core::StringVector possibleModes = {"Online", "EventReader"};
351  std::string runningMode;
352  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::XmlHelper::getAttribute(settingsElement,
353  "mode", runningMode, [&](const std::string &value){
354  return (std::find(possibleModes.begin(), possibleModes.end(), value) != possibleModes.end());
355  }));
356  if(runningMode == "Online") {
357  m_appRunningMode = ONLINE;
358  }
359  else {
360  m_appRunningMode = EVENT_READER;
361  }
362 
363  configureModule(moduleElement);
364  configureCycle(settingsElement);
365  configureNetwork(settingsElement);
366  configureEventReader(settingsElement);
367  configureArchiver(archiverElement);
368 
369  core::TiXmlHandle settingsHandle(settingsElement);
370  bool enableStatistics = false;
371  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND,!=,
372  core::XmlHelper::readParameter(settingsHandle, "StandaloneSleepTime", m_standaloneSleep));
373  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND,!=,
374  core::XmlHelper::readParameter(settingsHandle, "EnableStatistics", enableStatistics));
375  enableStats(enableStatistics);
376  }

380  void ModuleApplication::configureModule(core::TiXmlElement *element) {
381  if(m_moduleType.empty()) {
382  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::XmlHelper::getAttribute(element, "type", m_moduleType));
383  }
384  if(m_moduleName.empty()) {
385  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::XmlHelper::getAttribute(element, "name", m_moduleName));
386  }
387  if(m_moduleType.empty() or m_moduleName.empty()) {
388  dqm_error( "Undefined module type and/or module name (type: {0}, name: {1}).", m_moduleType, m_moduleName );
389  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
390  }
391  setName(m_moduleName);
392  m_module = core::PluginManager::instance()->create<Module>(m_moduleType);
393  if(nullptr == m_module) {
394  dqm_error( "Module '{0}' not registered in plugin manager!", m_moduleType );
395  dqm_error( "Please check your plugin settings (libraries, environment) and restart the application" );
396  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
397  }
398  if(nullptr != dynamic_cast<AnalysisModule*>(m_module.get())) {
399  m_appModuleType = ANALYSIS;
400  }
401  else if(nullptr != dynamic_cast<StandaloneModule*>(m_module.get())) {
402  m_appModuleType = STANDALONE;
403  }
404  else {
405  dqm_error( "Undefined module type, must be AnalysisModule or StandaloneModule" );
406  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
407  }
408  if(m_appRunningMode == EVENT_READER and m_appModuleType != ANALYSIS) {
409  dqm_error( "Application running in FileReader mode. Can run only module of type AnalysisModule" );
410  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
411  }
412  m_module->setModuleApplication(this);
413  const core::TiXmlHandle moduleHandle(element);
414 
415  m_allowBooking = true;
416  m_module->readSettings(moduleHandle);
417  m_module->initModule();
418  m_allowBooking = false;
419  }

423  void ModuleApplication::configureCycle(core::TiXmlElement *element) {
424  unsigned int timeout = 10;
425  unsigned int period = 30;
426  unsigned int counterLimit = 0;
427  if(nullptr != element) {
428  core::TiXmlHandle handle(element);
429  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CycleTimeout", timeout));
430  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CyclePeriod", period));
431  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CycleCounter", counterLimit));
432  }
433  if((timeout >= period) || (0 == period && 0 == counterLimit)) {
434  dqm_error( "Invalid cycle settings: period={0}, timeout={1}, counter={2}", period, timeout, counterLimit );
435  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
436  }
437  dqm_info( "== Cycle settings ==" );
438  dqm_info( "=> Period: {0}", period );
439  dqm_info( "=> Timeout: {0}", timeout );
440  dqm_info( "=> Counter: {0}", counterLimit );
441  dqm_info( "====================" );
442  m_cycle.setTimeout(timeout);
443  m_cycle.setTimerPeriod(period);
444  m_cycle.setCounterLimit(counterLimit);
445  }

497  void ModuleApplication::configureArchiver(core::TiXmlElement *element) {
498  if(nullptr == element) {
499  return;
500  }
501  bool enableArchiver = true;
502  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
503  core::XmlHelper::getAttribute(element, "enable", enableArchiver));
504  if(not enableArchiver) {
505  dqm_info( "The archiver has been disabled. No monitor element will be archived !" );
506  return;
507  }
508  core::TiXmlHandle handle(element);
509  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=,
510  core::XmlHelper::readParameter(handle, "FileName", m_inputArchiveName));
511  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
512  core::XmlHelper::readParameter(handle, "AllowOverwrite", m_allowOverwrite));
513  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
514  core::XmlHelper::readParameter(handle, "AppendRunNumber", m_archiverRunNumber));
515  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
516  core::XmlHelper::readParameter(handle, "WriteReferences", m_archiverWithReferences));
517  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
518  core::XmlHelper::readParameter(handle, "OpenMode", m_archiveOpenMode));
519  // finally create our archiver
520  m_archiver = std::make_shared<core::Archiver>();
521  auto selectorsElement = handle.FirstChildElement("selectors").Element();
522  if(nullptr != selectorsElement) {
523  for(auto selectorElement = selectorsElement->FirstChildElement("selector") ;
524  nullptr != selectorElement ; selectorElement = selectorElement->NextSiblingElement("selector")) {
525  std::string expression;
526  bool select = true;
527  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=,
528  core::XmlHelper::getAttribute(selectorElement, "regex", expression));
529  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=,
530  core::XmlHelper::getAttribute(selectorElement, "select", select));
531  // try to compile the regex first
532  try {
533  std::regex reg(expression);
534  }
535  catch(std::regex_error) {
536  dqm_error( "Archiver selector: invalid regular expression : {0}", expression );
537  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
538  }
539  dqm_debug( "Archiver append selector, expression: {0}, select: {1}", expression, select );
540  m_archiverSelector.addSelector([expression,select](core::MonitorElementPtr me)->bool{
541  core::Path path = me->path();
542  path += me->name();
543  std::regex pattern(expression);
544  const bool match = std::regex_match(path.getPath(), pattern);
545  return select ? match : not match;
546  });
547  }
548  }
549  m_archiver->setSelectorFunction(m_archiverSelector.function());
550  }

599  void ModuleApplication::receiveSubscriptionList(CommandEvent *cmd) {
600  core::json jsubscription = nullptr;
601  try {
602  jsubscription = core::json::parse(cmd->buffer().begin(), cmd->buffer().end());
603  }
604  catch(...) {
605  dqm_error( "Caught exception: Couldn't update monitor element subscription list !" );
606  return;
607  }
608  if(not jsubscription.is_array()) {
609  dqm_error( "Monitor element subscription json object is not a list !" );
610  return;
611  }
612  for(auto &element : jsubscription) {
613  auto path = element.value<std::string>("path", "");
614  auto eltName = element.value<std::string>("name", "");
615  auto subscribe = element.value<bool>("sub", false);
616 
617  OnlineElementPtr monitorElement = nullptr;
618  if(core::STATUS_CODE_SUCCESS != m_monitorElementManager->getMonitorElement(path, eltName, monitorElement)) {
619  dqm_warning( "While updating subscription list: couldn't find element path'{0}', name '{1}'", path, eltName );
620  continue;
621  }
622  monitorElement->setSubscribed(subscribe);
623  dqm_info( "Monitor element: path {0}, name {1} has been {2}subscribed", path, eltName, subscribe ? "" : "un-" );
624  }
625  }
