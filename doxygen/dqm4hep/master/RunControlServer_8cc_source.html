55  void RunControlServer::run() {
56  m_stopFlag = false;
57 
58  // create user external interface (plugin)
59  m_interface = PluginManager::instance()->create<RunControlInterface>(m_interfaceName);
60 
61  if( not m_interface ) {
62  dqm_error( "Couldn't find run control interface '{0}' in plugin manager", m_interfaceName );
63  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
64  }
65 
66  // create network interface
67  m_server = std::make_shared<dqm4hep::net::Server>(m_runControl.name());
68 
69  m_pSorService = m_server->createService(OnlineRoutes::RunControl::sor(m_runControl.name()));
70  m_runControl.onStartOfRun().connect(this, &RunControlServer::sor);
71 
72  m_pEorService = m_server->createService(OnlineRoutes::RunControl::eor(m_runControl.name()));
73  m_runControl.onEndOfRun().connect(this, &RunControlServer::eor);
74 
75  m_server->createRequestHandler(OnlineRoutes::RunControl::status(m_runControl.name()), this, &RunControlServer::sendCurrentRun);
76 
77  m_interface->setServer(this);
78  m_interface->readSettings(m_userParameters);
79 
80  // start the server
81  m_server->start();
82 
83  // run the user plugin
84  if(m_interface->runBlocking())
85  {
86  // block here until stop called from outside
87  m_interface->run();
88  }
89  else
90  {
91  // call run and sleep
92  m_interface->run();
93 
94  while( ! m_stopFlag )
95  core::time::msleep(1000);
96  }
97 
98  m_server->stop();
99  m_pSorService = nullptr;
100  m_pEorService = nullptr;
101  }
