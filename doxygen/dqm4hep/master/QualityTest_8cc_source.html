80  void QReportStorage::addReport(const QReport &qreport, bool warnOnReplace) {
81  const std::string &path(qreport.m_monitorElementPath);
82  const std::string &name(qreport.m_monitorElementName);
83  const std::string &qtname(qreport.m_qualityTestName);
84 
85  if (!warnOnReplace) {
86  QReportContainer::key_type key(path, name);
87  m_reports[key][qtname] = qreport;
88  } else {
89  QReportContainer::key_type key(path, name);
90  auto iter1 = m_reports.find(key);
91 
92  if (m_reports.end() == iter1)
93  iter1 = m_reports.insert(QReportContainer::value_type(key, QReportMap())).first;
94 
95  auto iter2 = iter1->second.find(qtname);
96 
97  if (iter1->second.end() == iter2) {
98  iter2 = iter1->second.insert(QReportMap::value_type(qtname, qreport)).first;
99  } else {
100  dqm_warning("QReportStorage::addReport: Replacing qreport path '{0}', name '{1}', qtest '{2}'", path, name,
101  qtname);
102  iter2->second = qreport;
103  }
104  }

303  void QualityTest::run(MonitorElement* monitorElement, QualityTestReport &report) {
304  this->fillBasicInfo(monitorElement, report);
305 
306  if (nullptr == monitorElement) {
307  report.m_message = "Couldn't run quality test: monitor element pointer is nullptr";
308  report.m_quality = 0.f;
309  report.m_qualityFlag = INVALID;
310  return;
311  }
312 
313  if (nullptr == monitorElement->object()) {
314  report.m_message = "Couldn't run quality test: ROOT monitor object is nullptr";
315  report.m_quality = 0.f;
316  report.m_qualityFlag = INVALID;
317  return;
318  }
319 
320  if(!this->enoughStatistics(monitorElement)) {
321  report.m_message = "Couldn't run quality test: Not enough statistics !";
322  report.m_quality = 0.f;
323  report.m_qualityFlag = INSUFFICENT_STAT;
324  return;
325  }
326 
327  try {
328  this->userRun(monitorElement, report);
329 
330  if(report.m_quality >= 0.f && report.m_quality < this->errorLimit()) {
331  report.m_qualityFlag = ERROR;
332  }
333  else if(report.m_quality >= this->errorLimit() && report.m_quality < this->warningLimit()) {
334  report.m_qualityFlag = WARNING;
335  }
336  else if(report.m_quality >= this->warningLimit() && report.m_quality <= 1.f) {
337  report.m_qualityFlag = SUCCESS;
338  }
339  else {
340  const std::string message("Quality value (" + typeToString(report.m_quality) + ") is out range !");
341 
342  if (!report.m_message.empty())
343  report.m_message += " " + message;
344  else
345  report.m_message = message;
346 
347  throw StatusCodeException(STATUS_CODE_OUT_OF_RANGE);
348  }
349  } catch (StatusCodeException &exception) {
350  const std::string message("Caught StatusCodeException while run QTest: " + exception.toString());
351 
352  if (!report.m_message.empty())
353  report.m_message += " " + message;
354  else
355  report.m_message = message;
356 
357  report.m_quality = 0.f;
358  report.m_qualityFlag = INVALID;
359  } catch (...) {
360  const std::string message("Caught unknown exception while run QTest");
361 
362  if (!report.m_message.empty())
363  report.m_message += " " + message;
364  else
365  report.m_message = message;
366 
367  report.m_quality = 0.f;
368  report.m_qualityFlag = INVALID;
369  }
370  }
