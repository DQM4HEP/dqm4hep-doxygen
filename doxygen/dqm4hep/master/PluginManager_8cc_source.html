92  StatusCode PluginManager::loadLibrary(const std::string &libraryName) {
93  size_t idx = libraryName.find_last_of("/");
94 
95  // the library basename, i.e. /path/to/libBlah.so --> libBlah.so
96  std::string libBaseName(libraryName.substr(idx + 1));
97 
98  char *real_path = realpath(libraryName.c_str(), nullptr);
99 
100  if (nullptr != real_path) {
101  dqm_info("<!-- Loading shared library : {0} ({1})-->", real_path, libBaseName);
102 
103  // use real_path
104  free(real_path);
105  } else {
106  dqm_info("<!-- Loading shared library : {0} ({1})-->", libraryName, libBaseName);
107  }
108 
109  void *pLibPointer = dlopen(libraryName.c_str(), RTLD_LAZY | RTLD_GLOBAL | RTLD_NODELETE);
110 
111  if (pLibPointer == nullptr) {
112  dqm_error("<!-- ERROR loading shared library : {0}\n -> {1} -->", libraryName, dlerror());
113  return STATUS_CODE_FAILURE;
114  }
115 
116  m_dlLibraries.push_back(pLibPointer);
117 
118  return STATUS_CODE_SUCCESS;
119  }

150  void PluginManager::registerPlugin(const Plugin *const plugin) {
151  // null ptr case
152  if (nullptr == plugin) {
153  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
154  }
155  // check if the plug is already registered
156  if (isPluginRegistered(plugin->name())) {
157  throw StatusCodeException(STATUS_CODE_ALREADY_PRESENT);
158  }
159  // try to register it
160  if (!m_pluginMap.insert(PluginMap::value_type(plugin->name(), plugin)).second) {
161  throw StatusCodeException(STATUS_CODE_FAILURE);
162  }
163  dqm_info("Registered new plugin {0}", plugin->name());
164  }
