186  StatusCode MonitorElementManager::createQualityTest(TiXmlElement *pXmlElement, float warning, float error) {
187  std::string name, type;
188  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(pXmlElement, "name", name));
189  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(pXmlElement, "type", type));
190 
191  float warningLimit(warning), errorLimit(error);
192 
193  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::getAttribute(pXmlElement, "error", errorLimit, [](const float &value){
194  return (value >= 0.f && value < 1.f);
195  }));
196 
197  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::getAttribute(pXmlElement, "warning", warningLimit, [&errorLimit](const float &value){
198  return (value >=0.f && value > errorLimit && value <= 1.f);
199  }));
200  // quality test already there ?
201  auto findIter = m_qualityTestMap.find(name);
202  if (m_qualityTestMap.end() != findIter) {
203  return STATUS_CODE_ALREADY_PRESENT;
204  }
205  // create our quality test from plugins
206  auto mgr = PluginManager::instance();
207  auto factory = mgr->create<QualityTestFactory>(type);
208  if(nullptr == factory) {
209  return STATUS_CODE_NOT_FOUND;
210  }
211  auto qualityTest = factory->createQualityTest(name);
212  // configure our quality test
213  qualityTest->setLimits(warningLimit, errorLimit);
214  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, qualityTest->readSettings(TiXmlHandle(pXmlElement)));
215  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, qualityTest->init());
216  // add it !
217  if (!m_qualityTestMap.insert(QualityTestMap::value_type(name, qualityTest)).second) {
218  return STATUS_CODE_FAILURE;
219  }
220  return STATUS_CODE_SUCCESS;
221  }
