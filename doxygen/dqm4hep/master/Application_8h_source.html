499  template <typename T>
500  inline void Application::sendStat(const std::string &entryName, const T &stats){
501  if(not statsEnabled() or noServer()) {
502  return;
503  }
504  core::json object = m_statistics[entryName];
505  if(object.is_null())
506  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
507  // add metadata on the fly
508  object["name"] = entryName;
509  object["value"] = stats;
510  object["appType"] = this->type();
511  object["appName"] = this->name();
512  object["time"] = core::time::asTime(core::time::now());
513  dqm_debug( "Sending app stat : \n'{0}'", object.dump() );
514  m_client.sendCommand(OnlineRoutes::OnlineManager::appStats(), object.dump());
515  }

555  inline void Application::directSubscribe(const std::string &serviceName, Controller *controller, void (Controller::*function)(const net::Buffer &)) {
556  auto findIter = m_serviceHandlerPtrMap.find(serviceName);
557 
558  if(m_serviceHandlerPtrMap.end() != findIter) {
559  dqm_error( "Application::directSubscribe(): couldn't subscribe twice to service '{0}'", serviceName );
560  throw core::StatusCodeException(core::STATUS_CODE_ALREADY_PRESENT);
561  }
562 
563  auto handler = std::make_shared<NetworkHandler>(m_eventLoop, serviceName, controller, function);
564  m_serviceHandlerPtrMap.insert(NetworkHandlerPtrMap::value_type(serviceName, handler));
565  m_client.subscribe(serviceName, handler.get(), &Application::NetworkHandler::sendServiceContent);
566  }
