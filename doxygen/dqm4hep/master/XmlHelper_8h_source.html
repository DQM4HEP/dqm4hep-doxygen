309  for (TiXmlElement *pXmlElement = xmlHandle.FirstChild("parameter").Element(); nullptr != pXmlElement;

403  StatusCode XmlHelper::parseParameterElement(const TiXmlElement *const element, std::string &name, T &value) {
404  if(nullptr == element) {
405  return STATUS_CODE_INVALID_PTR;
406  }
407  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(element, "name", name));
408  std::string valueStr;
409  if (STATUS_CODE_SUCCESS != XmlHelper::getAttribute(element, "value", valueStr)) {
410  valueStr = element->GetText() ? element->GetText() : "";
411  }
412  if (not dqm4hep::core::stringToType(valueStr, value)) {
413  return STATUS_CODE_FAILURE;
