106  void ExactRefCompareTest::userRun(MonitorElement* monitorElement, QualityTestReport &report) {
107 
108  const bool hasObject = (monitorElement->object() != nullptr);
109  const bool hasReference = (monitorElement->reference() != nullptr);
110 
111  if (!hasObject) {
112  report.m_message = "No object attached to monitor element";
113  report.m_quality = 0.f;
114  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
115  }
116 
117  if (!hasReference) {
118  report.m_message = "No reference attached to monitor element";
119  report.m_quality = 0.f;
120  throw StatusCodeException(STATUS_CODE_INVALID_PTR);
121  }
122 
123  const std::string objectClass(monitorElement->object()->ClassName());
124  const std::string referenceClass(monitorElement->reference()->ClassName());
125  const bool isHistogram = monitorElement->object()->InheritsFrom("TH1");
126  const bool isGraph1D = monitorElement->object()->InheritsFrom("TGraph");
127  const bool isGraph2D = monitorElement->object()->InheritsFrom("TGraph2D");
128  const bool sameType = (objectClass == referenceClass);
129 
130  if(!sameType) {
131  report.m_message = "Objet and reference do not have the same type";
132  report.m_quality = 0.f;
133  throw StatusCodeException(STATUS_CODE_FAILURE);
134  }
135 
136  if (isHistogram) {
137  this->doHistogramTest(monitorElement, report);
138  }
139  else if(isGraph1D) {
140  this->doGraph1DTest(monitorElement, report);
141  }
142  else if(isGraph2D) {
143  this->doGraph2DTest(monitorElement, report);
144  }
145  else {
146  this->doIsEqualTest(monitorElement, report);
147  }
148  }

152  void ExactRefCompareTest::doHistogramTest(MonitorElement* monitorElement, QualityTestReport &report) {
153 
154  TH1 *histogram = monitorElement->objectTo<TH1>();
155  TH1 *reference = monitorElement->referenceTo<TH1>();
156  int nBins(0), nBinsRef(0);
157 
158  if(histogram->GetDimension() == 1) {
159  nBins = histogram->GetNbinsX();
160  nBinsRef = reference->GetNbinsX();
161  }
162  else if(histogram->GetDimension() == 2) {
163  nBins = histogram->GetNbinsX() * histogram->GetNbinsY();
164  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY();
165  }
166  else if(histogram->GetDimension() == 3) {
167  nBins = histogram->GetNbinsX() * histogram->GetNbinsY() * histogram->GetNbinsZ();
168  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY() * reference->GetNbinsZ();
169  }
170  else {
171  report.m_message = "Histogram with invalid dimension (dim = " + typeToString(histogram->GetDimension()) + ") !";
172  report.m_quality = 0.f;
173  throw StatusCodeException(STATUS_CODE_FAILURE);
174  }
175 
176  if(nBins != nBinsRef) {
177  report.m_message = "Histogram and reference do not have the same number of bins";
178  report.m_quality = 0.f;
179  throw StatusCodeException(STATUS_CODE_FAILURE);
180  }
181 
182  const int firstBin = m_compareUnderflow ? 0 : 1;
183  const int lastBin = m_compareOverflow ? nBins : nBins+1;
184  unsigned int nDifferentBins(0);
185 
186  for(auto bin = firstBin ; bin <= lastBin ; ++bin) {
187 
188  const Double_t binContent(histogram->GetBinContent(bin));
189  const Double_t binContentRef(reference->GetBinContent(bin));
190  const bool binsAreDifferent(!(fabs(binContent-binContentRef) < std::numeric_limits<Double_t>::epsilon()));
191 
192  if(binsAreDifferent) {
193  ++nDifferentBins;
194  }
195  }
196 
197  if(nDifferentBins != 0) {
198  report.m_message = "Histogram and reference are different (difference = " + typeToString(nDifferentBins) + ")";
199  // Quality estimate as the percentage of different points.
200  // This factor is scaled to the error limit to make sure the quality flag is set to ERROR
201  report.m_quality = this->errorLimit() * (static_cast<float>(nBins) - static_cast<float>(nDifferentBins)) / static_cast<float>(nBins);
202  }
203  else {
204  report.m_message = "Histogram and reference are equal !";
205  report.m_quality = 1.f;
206  }
207 
208  report.m_extraInfos["nDifferences"] = nDifferentBins;
209  }

213  void ExactRefCompareTest::doGraph1DTest(MonitorElement* monitorElement, QualityTestReport &report) {
214 
215  TGraph *graph = monitorElement->objectTo<TGraph>();
216  TGraph *reference = monitorElement->referenceTo<TGraph>();
217  const unsigned int nPoints(graph->GetN());
218  const unsigned int nPointsRef(reference->GetN());
219 
220  if(nPoints != nPointsRef) {
221  report.m_message = "Graph and reference do not have the same number of points";
222  report.m_quality = 0.f;
223  throw StatusCodeException(STATUS_CODE_FAILURE);
224  }
225 
226  unsigned int nDifferentPoints(0);
227 
228  for(unsigned int point = 0 ; point < nPoints ; ++point) {
229  Double_t x, y, xRef, yRef;
230  graph->GetPoint(point, x, y);
231  reference->GetPoint(point, xRef, yRef);
232 
233  const bool xAreDifferent(!(fabs(x-xRef) < std::numeric_limits<Double_t>::epsilon()));
234  const bool yAreDifferent(!(fabs(y-yRef) < std::numeric_limits<Double_t>::epsilon()));
235 
236  if(xAreDifferent || yAreDifferent) {
237  ++nDifferentPoints;
238  }
239  }
240 
241  if(nDifferentPoints != 0) {
242  report.m_message = "Graph and reference are different (difference = " + typeToString(nDifferentPoints) + ")";
243  report.m_quality = this->errorLimit() * (static_cast<float>(nPoints) - static_cast<float>(nDifferentPoints)) / static_cast<float>(nPoints);
244  }
245  else {
246  report.m_message = "Graph and reference are equal !";
247  report.m_quality = 1.f;
248  }
249 
250  report.m_extraInfos["nDifferences"] = nDifferentPoints;
251  }
