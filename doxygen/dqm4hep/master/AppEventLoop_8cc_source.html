71  int AppEventLoop::exec() {
72  m_running = true;
73  m_quitFlag = false;
74 
75  m_timerThread = std::thread(&AppEventLoop::timerThread, this);
76 
77  while(1) {
78  if(m_quitFlag) {
79  break;
80  }
81 
82  // safely get the app event pointer
83  AppEvent* event = nullptr;
84 
85  {
86  std::lock_guard<std::recursive_mutex> lock(m_queueMutex);
87 
88  if(!m_eventQueue.empty()) {
89  event = *m_eventQueue.rbegin();
90  m_eventQueue.erase(event);
91  }
92  }
93 
94  // if no event, save cpu ressources ...
95  if(nullptr == event) {
96  usleep(100);
97  continue;
98  }
99 
100  try {
101  this->processEvent(event);
102  }
103  catch(...) {
104  m_returnCode = 1;
105  break;
106  }
107  delete event;
108  usleep(100);
109  }
110 
111  m_timerStopFlag = true;
112  m_timerThread.join();
113  m_running = false;
114  return m_returnCode.load();
115  }

200  void AppEventLoop::timerThread() {
201  {
202  // initialize start time point of all timers
203  std::lock_guard<std::recursive_mutex> lock(m_timerMutex);
204  for(auto timer : m_timers) {
205  timer->m_startTime = core::time::now();
206  }
207  }
208  while(not m_timerStopFlag.load()) {
209  {
210  std::lock_guard<std::recursive_mutex> lock(m_timerMutex);
211  for(auto timer : m_timers) {
212  if(timer->active()) {
213  const unsigned int timeoutEllapsed = std::chrono::duration_cast<std::chrono::milliseconds>(core::time::now()-timer->m_startTime).count();
214  const bool timeoutReached = (timeoutEllapsed >= timer->interval());
215  if(timeoutReached) {
216  // process timer timeout in event loop
217  processFunction([timer](){
218  timer->m_signal.emit();
219  });
220  // stop it if single shot
221  if(timer->singleShot()) {
222  timer->m_active = false;
223  }
224  else {
225  timer->m_startTime = core::time::now();
226  }
227  }
228  }
229  if(m_timerStopFlag.load()) {
230  break;
231  }
232  }
233  }
234  usleep(100);
235  }
236  dqm_debug( "Exiting timer thread !" );
237  }
