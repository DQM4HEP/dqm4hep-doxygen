140  void EventCollector::handleRegistration(const net::Buffer &request, net::Buffer &response) {
141  core::json registrationDetails({});
142  if(0 != request.size()) {
143  registrationDetails = core::json::parse(request.begin(), request.end());
144  }
145  auto clientSourceName = registrationDetails.value<std::string>("source", "");
146  auto clientId = this->serverClientId();
147  auto findIter = m_sourceInfoMap.find(clientSourceName);
148  core::json clientResponseValue({});
149 
150  // source already registered
151  if(m_sourceInfoMap.end() != findIter) {
152  if(clientId == findIter->second.m_clientId) {
153  clientResponseValue["message"] = "Event source already registered !";
154  clientResponseValue["registered"] = true;
155  }
156  else {
157  std::stringstream ss; ss << "Event source already registered with a different client ID (" << findIter->second.m_clientId << ") !";
158  clientResponseValue["message"] = ss.str();
159  clientResponseValue["registered"] = false;
160  }
161  }
162  // source not registered yet
163  else {
164  std::string sourceName = registrationDetails.value<std::string>("source", "");
165  SourceInfo sourceInfo;
166  findIter = m_sourceInfoMap.insert(SourceInfoMap::value_type(sourceName, std::move(sourceInfo))).first;
167 
168  findIter->second.m_clientId = clientId;
169  findIter->second.m_name = registrationDetails.value<std::string>("source", "");
170  findIter->second.m_streamerName = registrationDetails.value<std::string>("streamer", "");
171  findIter->second.m_eventService = createService(OnlineRoutes::EventCollector::eventUpdate(name(), findIter->first));
172 
173  auto collectors = registrationDetails["collectors"];
174  auto hostInfo = registrationDetails["host"];
175 
176  for(auto collector : collectors) {
177  findIter->second.m_collectors.push_back(collector.get<std::string>());
178  }
179 
180  for(auto hostInfoIter : hostInfo.items()) {
181  findIter->second.m_hostInfo[hostInfoIter.key()] = hostInfo[hostInfoIter.key()].get<std::string>();
182  }
183 
184  auto model = findIter->second.m_buffer.createModel<std::string>();
185  findIter->second.m_buffer.setModel(model);
186 
187  dqm_info( "New event source '{0}' registered with client id {1}", findIter->second.m_name, findIter->second.m_clientId );
188 
189  clientResponseValue["registered"] = true;
190  sendStat("NSources", m_sourceInfoMap.size());
191  }
192 
193  auto model = response.createModel<std::string>();
194  model->copy(clientResponseValue.dump());
195  response.setModel(model);
196  }

200  void EventCollector::handleClientExit(StoreEvent<int> *event) {
201  const int clientId(event->data());
202  auto findIter = std::find_if(m_sourceInfoMap.begin(), m_sourceInfoMap.end(), [&clientId](const SourceInfoMap::value_type &iter){
203  return (iter.second.m_clientId == clientId);
204  });
205 
206  if(findIter != m_sourceInfoMap.end()) {
207  dqm_info( "Removing event source '{0}' from source list !", findIter->second.m_name );
208  m_sourceInfoMap.erase(findIter);
209  sendStat("NSources", m_sourceInfoMap.size());
210  }
211  }

215  void EventCollector::handleCollectEvent(const net::Buffer &buffer) {
216  const int clientId(serverClientId());
217 
218  auto findIter = std::find_if(m_sourceInfoMap.begin(), m_sourceInfoMap.end(), [&clientId](const SourceInfoMap::value_type &iter){
219  return (iter.second.m_clientId == clientId);
220  });
221 
222  if(findIter != m_sourceInfoMap.end()) {
223  std::string copiedBuffer(buffer.begin(), buffer.size());
224  auto newModel = findIter->second.m_buffer.createModel<std::string>();
225  findIter->second.m_buffer.setModel(newModel);
226  newModel->move(std::move(copiedBuffer));
227 
228  m_nCollectedEvents10++;
229  m_nCollectedEvents60++;
230  m_nCollectedBytes10 += buffer.size();
231  m_nCollectedBytes60 += buffer.size();
232  // send update
233  findIter->second.m_eventService->sendBuffer(buffer.begin(), buffer.size());
234  }
235  }
