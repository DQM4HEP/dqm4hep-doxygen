156  void EventCollector::handleRegistration(const net::Buffer &request, net::Buffer &response) {
157  core::json registrationDetails({});
158  if(0 != request.size()) {
159  registrationDetails = core::json::parse(request.begin(), request.end());
160  }
161  auto clientSourceName = registrationDetails.value<std::string>("source", "");
162  auto clientId = this->serverClientId();
163  auto findIter = m_sourceInfoMap.find(clientSourceName);
164  core::json clientResponseValue({});
165 
166  // source already registered
167  if(m_sourceInfoMap.end() != findIter) {
168  if(clientId == findIter->second.m_clientId) {
169  clientResponseValue["message"] = "Event source already registered !";
170  clientResponseValue["registered"] = true;
171  }
172  else {
173  std::stringstream ss; ss << "Event source already registered with a different client ID (" << findIter->second.m_clientId << ") !";
174  clientResponseValue["message"] = ss.str();
175  clientResponseValue["registered"] = false;
176  }
177  }
178  // source not registered yet
179  else {
180  std::string sourceName = registrationDetails.value<std::string>("source", "");
181  SourceInfo sourceInfo;
182  findIter = m_sourceInfoMap.insert(SourceInfoMap::value_type(sourceName, std::move(sourceInfo))).first;
183 
184  findIter->second.m_clientId = clientId;
185  findIter->second.m_name = registrationDetails.value<std::string>("source", "");
186  findIter->second.m_streamerName = registrationDetails.value<std::string>("streamer", "");
187  findIter->second.m_eventService = createService(OnlineRoutes::EventCollector::eventUpdate(name(), findIter->first));
188 
189  auto collectors = registrationDetails["collectors"];
190  auto hostInfo = registrationDetails["host"];
191 
192  for(auto collector : collectors) {
193  findIter->second.m_collectors.push_back(collector.get<std::string>());
194  }
195 
196  for(auto hostInfoIter : hostInfo.items()) {
197  findIter->second.m_hostInfo[hostInfoIter.key()] = hostInfo[hostInfoIter.key()].get<std::string>();
198  }
199 
200  auto model = findIter->second.m_buffer.createModel<std::string>();
201  findIter->second.m_buffer.setModel(model);
202 
203  dqm_info( "New event source '{0}' registered with client id {1}", findIter->second.m_name, findIter->second.m_clientId );
204 
205  clientResponseValue["registered"] = true;
206  sendStat("NSources", m_sourceInfoMap.size());
207  }
208 
209  auto model = response.createModel<std::string>();
210  model->copy(clientResponseValue.dump());
211  response.setModel(model);
212  }

216  void EventCollector::handleClientExit(StoreEvent<int> *event) {
217  const int clientId(event->data());
218  auto findIter = std::find_if(m_sourceInfoMap.begin(), m_sourceInfoMap.end(), [&clientId](const SourceInfoMap::value_type &iter){
219  return (iter.second.m_clientId == clientId);
220  });
221 
222  if(findIter != m_sourceInfoMap.end()) {
223  dqm_info( "Removing event source '{0}' from source list !", findIter->second.m_name );
224  m_sourceInfoMap.erase(findIter);
