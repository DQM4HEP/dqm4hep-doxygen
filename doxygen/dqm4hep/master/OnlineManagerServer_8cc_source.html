62  void OnlineManagerServer::collectLog(const net::Buffer &buffer) {
63  core::json message;
64  try {
65  message = core::json::parse(buffer.begin(), buffer.end());
66  }
67  catch(core::json::parse_error& e) {
68  dqm_warning( "OnlineManagerServer::collectLog: received unexpected log message (json parse error)" );
69  return;
70  }
71  // check log consistency
72  bool logConsistent = (
73  1 == message.count("logger") &&
74  1 == message.count("level") &&
75  1 == message.count("message") &&
76  1 == message.count("host") &&
77  1 == message.count("pid")
78  );
79  if(not logConsistent) {
80  dqm_warning( "OnlineManagerServer::collectLog: received inconsistent log message" );
81  }
82  if(nullptr != m_logger) {
83  this->logMessage(message);
84  }
85  // send the log message to listeners
86  m_logsService->sendBuffer(buffer.begin(), buffer.size());
87  }
