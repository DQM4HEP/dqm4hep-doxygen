220  int Application::exec() {
221  m_running = true;
222  if(not noServer()) {
223  m_server->start();
224  sleep(1);
225  }
226  this->setState("Running");
227 
228  try {
229  dqm_info( "Application on start ..." );
230  this->onStart();
231  }
232  catch(...) {
233  dqm_error( "Application::init(): failed to start the app !" );
234  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
235  }
236 
237  int returnCode(0);
238 
239  try {
240  if(statsEnabled() and not noServer()) {
241  // send statistics every 5 seconds
242  dqm_info( "Statistics enabled. Creating timer (5 secs) ..." );
243  m_appStatTimer = createTimer();
244  m_appStatTimer->setInterval(5000); // 5 seconds
245  m_appStatTimer->setSingleShot(false);
246  m_appStatTimer->onTimeout().connect(this, &Application::sendAppStats);
247  m_appStatTimer->start();
248  }
249  m_eventLoop.connectOnEvent(this, &Application::onEvent);
250  dqm_info( "Starting event loop ..." );
251  returnCode = m_eventLoop.exec();
252  dqm_info( "Exiting event loop ..." );
253  m_eventLoop.disconnectOnEvent(this);
254  if(not noServer()){
255  m_server->onClientExit().disconnect();
256  }
257  this->onStop();
258  }
259  catch(...) {
260  dqm_error( "Application::init(): failed to start the app !" );
261  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
262  }
263  std::stringstream appState; appState << "Exiting (" << returnCode << ")";
264  this->setState(appState.str());
265  m_running = false;
266 
267  return returnCode;
268  }

363  void Application::sendAppStats() {
364  core::MemoryStats memInfo;
365  core::memStats(memInfo);
366  core::procStats(m_stats);
367 
368  core::dqm_long procVm = m_stats.vm/1024.; // in MB
369  core::dqm_long procRss = m_stats.rss/1024.; // in MB
370 
371  dqm_debug("Sending internal app stats ..." );
372  sendStat("VmProc", procVm);
373  sendStat("VmTotal", (procVm/(memInfo.vmTot*1.))*100.);
374  sendStat("VmInUse", (procVm/(memInfo.vmUsed*1.))*100.);
375 
376  sendStat("RSSProc", procRss);
377  sendStat("RSSTotal", (procRss/(memInfo.rssTot*1.))*100.);
378  sendStat("RSSInUse", (procRss/(memInfo.rssUsed*1.))*100.);
379 
380  sendStat("CpuUser", m_stats.cpuUser);
381  sendStat("CpuSys", m_stats.cpuSys);
382  sendStat("CpuTotal", m_stats.cpuTot);
383 
384  auto tm_time = localtime(&m_stats.lastPollTime.tv_sec);
385  char date_buf[100];
386  std::strftime(date_buf, sizeof(date_buf), "%Y-%m-%d %H:%M:%S", tm_time);
387 
388  sendStat("LastUpdate", date_buf );
389  dqm_debug( "Sending internal app stats ... OK" );

452  void Application::NetworkHandler::postServiceContent(const net::Buffer &buffer) {
453  int count = m_eventLoop.count([this](std::shared_ptr<AppEvent> ptr){
454  ServiceUpdateEvent *evt = dynamic_cast<ServiceUpdateEvent*>(ptr.get());
455  return (evt && evt->serviceName() == this->m_name);
456  });
457 
458  if(count >= m_maxNEvents) {
459  dqm_debug( "NetworkHandler::postServiceContent(): maximum of posted service updates reached ({0}) !", m_maxNEvents );
460  return;
461  }
462 
463  // copy buffer content
464  std::string content;
465  content.assign(buffer.begin(), buffer.size());
466 
467  // create new buffer model
468  auto bufferModel = buffer.createModel<std::string>();
469  bufferModel->move(std::move(content));
470 
471  // create the event to post, pass the copied buffer in ctor
472  ServiceUpdateEvent *pEvent = new ServiceUpdateEvent(m_name, bufferModel);
473  pEvent->setPriority(m_priority);
474 
475  // post the event !
476  m_eventLoop.postEvent(pEvent);
477  }
