207  m_server->start();
208  sleep(1);
209  this->setState("Running");
210 
211  try {
212  dqm_info( "Application on start ..." );
213  this->onStart();
214  }
215  catch(...) {
216  dqm_error( "Application::init(): failed to start the app !" );
217  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
218  }
219 
220  int returnCode(0);
221 
222  try {
223  if(m_statsEnabled) {
224  // send statistics every 5 seconds
225  dqm_info( "Statistics enabled. Creating timer (5 secs) ..." );
226  m_eventLoop.createTimer("AppStats", 5, false, this, &Application::sendAppStats);
227  }
228  m_eventLoop.connectOnEvent(this, &Application::onEvent);
229  dqm_info( "Starting event loop ..." );
230  returnCode = m_eventLoop.exec();
231  dqm_info( "Exiting event loop ..." );
232  m_eventLoop.disconnectOnEvent(this);
233  m_server->onClientExit().disconnectAll();
234  this->onStop();
235  }
236  catch(...) {
237  dqm_error( "Application::init(): failed to start the app !" );
238  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
239  }
240  std::stringstream appState; appState << "Exiting (" << returnCode << ")";
241  this->setState(appState.str());
242  m_running = false;
243 
244  return returnCode;
245  }

370  void Application::NetworkHandler::postServiceContent(const net::Buffer &buffer) {
371  int count = m_eventLoop.count([this](std::shared_ptr<AppEvent> ptr){
372  ServiceUpdateEvent *evt = dynamic_cast<ServiceUpdateEvent*>(ptr.get());
373  return (evt && evt->serviceName() == this->m_name);
374  });
375 
376  if(count >= m_maxNEvents) {
377  dqm_debug( "NetworkHandler::postServiceContent(): maximum of posted service updates reached ({0}) !", m_maxNEvents );
378  return;
379  }
380 
381  // copy buffer content
382  std::string content;
383  content.assign(buffer.begin(), buffer.size());
384 
385  // create new buffer model
386  auto bufferModel = buffer.createModel<std::string>();
387  bufferModel->move(std::move(content));
388 
389  // create the event to post, pass the copied buffer in ctor
390  ServiceUpdateEvent *pEvent = new ServiceUpdateEvent(m_name, bufferModel);
391  pEvent->setPriority(m_priority);
392 
393  // post the event !
394  m_eventLoop.postEvent(pEvent);
395  }
