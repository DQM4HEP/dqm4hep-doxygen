94  if(this->isInitialized())

96 
97  size_t splitterPosition = settingsFileName.find(":");
98 
99  std::string fileHandlerType;
100  std::string filePattern;
101 
102  if(splitterPosition != std::string::npos)
103  {
104  fileHandlerType = settingsFileName.substr(0, splitterPosition);
105  filePattern = settingsFileName.substr(splitterPosition+1);
106  }
107  else
108  {
109  fileHandlerType = "";
110  filePattern = settingsFileName;
111  }
112 
113  LOG4CXX_INFO( dqmMainLogger , "Starting analysis module application with file handler type " << fileHandlerType );
114 
115  FileHandlerFactory fileHandlerFactory;
116  FileHandler *pFileHandler = fileHandlerFactory.createFileHandler(fileHandlerType);
117 
118  if(!pFileHandler)
119  return STATUS_CODE_FAILURE;
120 
121  StatusCode statusCode = pFileHandler->download(filePattern);
122 
123  if(statusCode != STATUS_CODE_SUCCESS)
124  {
125  delete pFileHandler;
126  return statusCode;
127  }
128 
129  std::string localSettingsFile = pFileHandler->getLocalFileName();
130  delete pFileHandler;
131 
132  TiXmlDocument xmlDocument(localSettingsFile);
133 
134  if (!xmlDocument.LoadFile())
135  {
136  LOG4CXX_FATAL( dqmMainLogger , "AnalysisModuleApplication::readSettings - Invalid xml file." );
137  return STATUS_CODE_FAILURE;
138  }
139 
140  const TiXmlHandle xmlDocumentHandle(&xmlDocument);
141 
142  if( ! m_replacementParameters.empty() )
143  {
144  TiXmlElement *pXmlElement = xmlDocumentHandle.FirstChildElement().Element();
145  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::replaceAllXmlAttributes(pXmlElement, m_replacementParameters));
146  }
147 
148  const TiXmlHandle xmlHandle = TiXmlHandle(xmlDocumentHandle.FirstChildElement().Element());
149 
150  // configure module
151  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, configureModule(xmlHandle));
152

154  int nServers = browser.getServers();
155 
156  std::string moduleServerName = "DQM4HEP/Module/" + this->getModule()->getName();

158

169  << "Please, change the module name or stop the other module application with the same name !" );

171  }
172  }
173  }
174 
175  DimServer::start( moduleServerName.c_str() );
176 
177  // configure remaining part of the application
178  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->configureCycle(xmlHandle));
179  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->configureArchiver(xmlHandle));
180  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->configureNetwork(xmlHandle));
181 
182  m_settings.m_settingsFileName = settingsFileName;
183  m_settings.print(m_moduleLogStr);
184 
185  setInitialized(true);
186

199  StatusCode AnalysisModuleApplication::run()
200  {
201  if(!this->isInitialized())
202  return STATUS_CODE_NOT_INITIALIZED;
203 
204  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->startServices());
205 
206  // get casted module for easier manipulation
207  AnalysisModule *pAnalysisModule = dynamic_cast<AnalysisModule *>(this->getModule());
208  std::string moduleName = pAnalysisModule->getName();
209 
210  // infinite loop
211  while(!this->shouldStopApplication())
212  {
213  // wait for a start of run
214  bool firstMessage = true;
215 
216  while(!m_pRunControlClient->isRunning() && !this->shouldStopApplication())
217  {
218  if(firstMessage)
219  {
220  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , waiting for next run ..." );
221  firstMessage = false;
222  }
223 
224  CoreTool::sleep(std::chrono::seconds(1));
225  }
226 
227  if(this->shouldStopApplication())
228  break;
229 
230  Run *pRun = m_pRunControlClient->getCurrentRun();
231 
232  if(NULL == pRun)
233  continue;
234 
235  m_runNumber = pRun->getRunNumber();
236  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Starting new run no " << m_runNumber );
237 
238  // open an archive
239  if(m_settings.m_shouldOpenArchive)
240  {
241  std::stringstream archiveFileName;
242  std::string directory =
243  m_settings.m_archiveDirectory.at(m_settings.m_archiveDirectory.size() - 1) == '/' ?
244  m_settings.m_archiveDirectory : m_settings.m_archiveDirectory + "/";
245 
246  archiveFileName << directory
247  << "DQMArchive"
248  << "_M" << moduleName
249  << "_I" << pRun->getRunNumber();
250 
251  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_pArchiver->open(archiveFileName.str(), "RECREATE"));
252  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Archive '" << m_pArchiver->getFileName() << "' opened" );
253  }
254 
255  // start of run !
256  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, pAnalysisModule->startOfRun(pRun));
257 
258  // start receiving data
259  m_pEventClient->setUpdateMode(true);
260 
261  // while run has not ended, process cycles
262  while( m_pRunControlClient->isRunning() && !this->shouldStopApplication() )
263  {
264  // process a cycle
265  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Start of cycle");
266  m_pCycle->startCycle();
267 
268  while(1)
269  {
270  if( m_pCycle->isEndOfCycleReached() )
271  break;
272 
273  if( m_pCycle->isTimeoutReached() )
274  break;
275 
276  if( this->shouldStopApplication() )
277  break;
278 
279  EventPtr event;
280  m_pEventClient->takeEvent(event);
281 
282  if(NULL == event)
283  continue;
284 
285  try
286  {
287  StatusCode statusCode = pAnalysisModule->processEvent(event.get());
288  m_pCycle->eventProcessed(event.get());
289  }
290  catch(StatusCodeException &exception)
291  {
292  LOG4CXX_ERROR( dqmMainLogger , m_moduleLogStr << " , Module::processEvent(evt) returned " << exception.toString() << " !");
293  }
294  catch(...)
295  {
296  LOG4CXX_ERROR( dqmMainLogger , m_moduleLogStr << " , Module::processEvent(evt) caught unknown exception !");
297  }
298  }
299 
300  m_pCycle->stopCycle();
301 
302  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , End of cycle reached !");
303  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Printing statistics :");
304  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , *** N processed events : " << m_pCycle->getNProcessedEvents());
305  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , *** Event rate : " << m_pCycle->getProcessingRate() << " evts/s");
306  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , *** Processing time : " << m_pCycle->getTotalCycleTime().count() << " s");
307 
308  // archive publication if user asked for
309  if(m_settings.m_shouldOpenArchive)
310  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_pArchiver->archive(pAnalysisModule));
311 
312  // MonitorElementList monitorElementListToPublish;
313  // RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->getMonitorElementManager()
314  // ->getMonitorElementListToPublish(monitorElementListToPublish));
315 
316  // TODO find a way to update the run number in monitor elements
317  // // set monitor elements infos
318  // for(MonitorElementList::iterator iter = monitorElementListToPublish.begin(), endIter = monitorElementListToPublish.end() ;
319  // endIter != iter ; ++iter)
320  // (*iter)->setRunNumber(this->getCurrentRunNumber());
321 
322  // send monitor elements to collector
323  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->getMonitorElementSender()->sendMonitorElements());
324 
325  // reset the monitor elements with end of run reset policy
326  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->getMonitorElementManager()
327  ->resetMonitorElements(END_OF_CYCLE_RESET_POLICY));
328 
329  // stop current run if run number has changed or run stopped
330  if(m_pRunControlClient->getCurrentRunNumber() != this->getCurrentRunNumber())
331  break;
332  }
333 
334  // stop receive data and clear contents
335  m_pEventClient->setUpdateMode(false);
336  m_pEventClient->clearQueue();
337 
338  // fill the run end time
339  pRun->setEndTime(CoreTool::now());
340 
341  // process the end of run module stuff
342  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, pAnalysisModule->endOfRun(pRun));
343 
344  // reset the monitor elements that have been reset at end of run
345  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->getMonitorElementManager()
346  ->resetMonitorElements(END_OF_RUN_RESET_POLICY));
347 
348  // archive the monitor element if archive is opened
349  if(m_settings.m_shouldOpenArchive)
350  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_pArchiver->close());
351  }
352 
353  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, stopServices());
354 
355  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Exiting application !");
356 
357  return this->getReturnCode();
358  }

390  StatusCode AnalysisModuleApplication::configureNetwork(const TiXmlHandle xmlHandle)
391  {
392  try
393  {
394  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Configuring network ...");
395 
396  //
397  // Get all xml attributes needed to configure network options !
398  //
399 
400  TiXmlElement *const pXmlElement = xmlHandle.FirstChildElement("network").Element();
401 
402  if(NULL == pXmlElement)
403  {
404  LOG4CXX_FATAL( dqmMainLogger , m_moduleLogStr << " , <network> element not found !");
405  return STATUS_CODE_NOT_FOUND;
406  }
407 
408  const TiXmlHandle networkHandle(pXmlElement);
409 
410  TiXmlElement *const pRunControlXmlElement = networkHandle.FirstChildElement("runcontrol").Element();
411  TiXmlElement *const pEventCollectorXmlElement = networkHandle.FirstChildElement("eventcollector").Element();
412  TiXmlElement *const pMonitorElementCollectorXmlElement = networkHandle.FirstChildElement("monitorelementcollector").Element();
413 
414  if(NULL == pRunControlXmlElement
415  || NULL == pEventCollectorXmlElement
416  || NULL == pMonitorElementCollectorXmlElement)
417  {
418  LOG4CXX_FATAL( dqmMainLogger , m_moduleLogStr << " , incomplete network xml element !");
419  return STATUS_CODE_NOT_FOUND;
420  }
421 
422  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(pRunControlXmlElement, "name", m_settings.m_runControlName));
423  THROW_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::getAttribute(pRunControlXmlElement, "type", m_settings.m_runControlType));
424  THROW_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::getAttribute(pEventCollectorXmlElement, "type", m_settings.m_eventClientType));
425  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(pMonitorElementCollectorXmlElement, "name", m_settings.m_monitorElementCollector));
426 
427  //
428  // Configure network form the extracted xml values
429  //
430 
431  // monitor element collector name
432  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->getMonitorElementSender()->setCollectorName(m_settings.m_monitorElementCollector));
433 
434  // event client
435  EventClient *pEventClient = PluginManager::instance()->createPluginClass<EventClient>(m_settings.m_eventClientType);
436 
437  if( ! pEventClient )
438  throw StatusCodeException(STATUS_CODE_FAILURE);
439 
440  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, pEventClient->readSettings( TiXmlHandle(pEventCollectorXmlElement) ));
441  // force update mode to false for the moment
442  pEventClient->setUpdateMode(false);
443  m_pEventClient = pEventClient;
444 
445  // configure run control
446  RunControlClient *pRunControlClient = PluginManager::instance()->createPluginClass<RunControlClient>(m_settings.m_runControlType);
447 
448  if(NULL == pRunControlClient)
449  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
450 
451  pRunControlClient->setRunControlName( m_settings.m_runControlName );
452  m_pRunControlClient = pRunControlClient;
453 
454  LOG4CXX_INFO( dqmMainLogger , m_moduleLogStr << " , Configuring network ... OK");
455  }
456  catch(const StatusCodeException &exception)
457  {
458  return exception.getStatusCode();
459  }
460 
461  return STATUS_CODE_SUCCESS;
462  }
