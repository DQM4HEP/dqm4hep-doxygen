143  void ModuleApplication::onEvent(AppEvent * appEvent) {
144  if(AppEvent::SERVICE_UPDATE == appEvent->type()) {
145  ServiceUpdateEvent* svc = dynamic_cast<ServiceUpdateEvent*>(appEvent);
146  // Start of run
147  if(svc->serviceName() == OnlineRoutes::RunControl::sor(m_runControl.name())) {
148  processStartOfRun(svc);
149  }
150  // End of run
151  if(svc->serviceName() == OnlineRoutes::RunControl::eor(m_runControl.name())) {
152  processEndOfRun();
153  }
154  }
155  if(AppEvent::COMMAND_HANDLING == appEvent->type()) {
156  CommandEvent *cmd = dynamic_cast<CommandEvent*>(appEvent);
157  // update subscription list
158  if(cmd->commandName() == OnlineRoutes::ModuleApplication::subscribe(name())) {
159  receiveSubscriptionList(cmd);
160  }
161  }
162  if(AppEvent::END_OF_RUN == appEvent->type()) {
163  auto eorEvent = dynamic_cast<StoreEvent<core::Run>*>(appEvent);
164  auto run = eorEvent->data();
165  m_runControl.endCurrentRun(run.parameters());
166  }
167  // process event received from the event collector
168  if(AppEvent::PROCESS_EVENT == appEvent->type() and ModuleApplication::ANALYSIS == m_mode) {
169  if(m_runControl.isRunning()) {
170  auto procEvent = dynamic_cast<StoreEvent<core::EventPtr>*>(appEvent);
171  auto anaModule = moduleAs<AnalysisModule>();
172  anaModule->process(procEvent->data());
173  m_cycle.incrementCounter();
174  }
175  }
176  // generic loop
177  if(AppEvent::PROCESS_EVENT == appEvent->type() and ModuleApplication::STANDALONE == m_mode) {
178  auto standModule = moduleAs<StandaloneModule>();
179  standModule->process();
180  m_cycle.incrementCounter();
181  m_standaloneTimer->start();
182  }
183  // end of cycle
184  if(AppEvent::END_OF_CYCLE == appEvent->type()) {
185  auto eocEvent = dynamic_cast<StoreEvent<EOCCondition>*>(appEvent);
186  auto condition = eocEvent->data();
187  m_module->endOfCycle(condition);
188  if(condition.m_counter > 0) {
189  try {
190  core::QReportStorage reportStorage;
191  // process quality tests
192  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_monitorElementManager->runQualityTests(reportStorage));
193  core::json jreports;
194  reportStorage.toJson(jreports);
195  dqm_info( jreports.dump(2) );
196 
197  OnlineElementPtrList publishElements;
198  m_monitorElementManager->iterate<OnlineElement>([&](OnlineElementPtr monitorElement){
199  if(not monitorElement->publish() or not monitorElement->subscribed()){
200  return true;
201  }
202  publishElements.push_back(monitorElement);
203  return true;
204  });
205  // TODO send monitor element to collector
206  }
207  catch(core::StatusCodeException &exception) {
208  dqm_error( "Error caught at end of cycle: {0}", exception.getStatusCode() );
209  }
210  }
211  // always restart a new cycle for standalone modules
212  if(ModuleApplication::STANDALONE == m_mode) {
213  m_module->startOfCycle();
214  m_cycle.startCycle(true);
215  }
216  // forced end is a synonym of end of run for analysis modules
217  // do not restart a cycle in this case
218  if(ModuleApplication::ANALYSIS == m_mode and not condition.m_forcedEnd) {
219  m_module->startOfCycle();
220  m_cycle.startCycle(true);
221  }
222  }
223  }

329  void ModuleApplication::configureModule(core::TiXmlElement *element) {
330  if(m_moduleType.empty()) {
331  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::XmlHelper::getAttribute(element, "type", m_moduleType));
332  }
333  if(m_moduleName.empty()) {
334  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, core::XmlHelper::getAttribute(element, "name", m_moduleName));
335  }
336  if(m_moduleType.empty() or m_moduleName.empty()) {
337  dqm_error( "Undefined module type and/or module name (type: {0}, name: {1}).", m_moduleType, m_moduleName );
338  throw core::StatusCodeException(core::STATUS_CODE_FAILURE);
339  }
340  setName(m_moduleName);
341  m_module = core::PluginManager::instance()->create<Module>(m_moduleType);
342  if(nullptr == m_module) {
343  dqm_error( "Module '{0}' not registered in plugin manager!", m_moduleType );
344  dqm_error( "Please check your plugin settings (libraries, environment) and restart the application" );
345  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
346  }
347  if(nullptr != dynamic_cast<AnalysisModule*>(m_module.get())) {
348  m_mode = ANALYSIS;
349  }
350  else if(nullptr != dynamic_cast<StandaloneModule*>(m_module.get())) {
351  m_mode = STANDALONE;
352  }
353  else {
354  dqm_error( "Undefined module type, must be AnalysisModule or StandaloneModule" );
355  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
356  }
357  m_module->setModuleApplication(this);
358  const core::TiXmlHandle moduleHandle(element);
359 
360  m_allowBooking = true;
361  m_module->readSettings(moduleHandle);
362  m_module->initModule();
363  m_allowBooking = false;
364  }

368  void ModuleApplication::configureCycle(core::TiXmlElement *element) {
369  unsigned int timeout = 10;
370  unsigned int period = 30;
371  unsigned int counterLimit = 0;
372  if(nullptr != element) {
373  core::TiXmlHandle handle(element);
374  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CycleTimeout", timeout));
375  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CyclePeriod", period));
376  THROW_RESULT_IF_AND_IF(core::STATUS_CODE_SUCCESS, core::STATUS_CODE_NOT_FOUND, !=, core::XmlHelper::readParameter(handle, "CycleCounter", counterLimit));
377  }
378  if((timeout >= period) || (0 == period && 0 == counterLimit)) {
379  dqm_error( "Invalid cycle settings: period={0}, timeout={1}, counter={2}", period, timeout, counterLimit );
380  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PARAMETER);
381  }
382  dqm_info( "== Cycle settings ==" );
383  dqm_info( "=> Period: {0}", period );
384  dqm_info( "=> Timeout: {0}", timeout );
385  dqm_info( "=> Counter: {0}", counterLimit );
386  dqm_info( "====================" );
387  m_cycle.setTimeout(timeout);
388  m_cycle.setTimerPeriod(period);
389  m_cycle.setCounterLimit(counterLimit);
390  }

459  void ModuleApplication::receiveSubscriptionList(CommandEvent *cmd) {
460  core::json jsubscription = nullptr;
461  try {
462  jsubscription = core::json::parse(cmd->buffer().begin(), cmd->buffer().end());
463  }
464  catch(...) {
465  dqm_error( "Caught exception: Couldn't update monitor element subscription list !" );
466  return;
467  }
468  if(not jsubscription.is_array()) {
469  dqm_error( "Monitor element subscription json object is not a list !" );
470  return;
471  }
472  for(auto &element : jsubscription) {
473  auto path = element.value<std::string>("path", "");
474  auto eltName = element.value<std::string>("name", "");
475  auto subscribe = element.value<bool>("sub", false);
476 
477  OnlineElementPtr monitorElement = nullptr;
478  if(core::STATUS_CODE_SUCCESS != m_monitorElementManager->getMonitorElement(path, eltName, monitorElement)) {
479  dqm_warning( "While updating subscription list: couldn't find element path'{0}', name '{1}'", path, eltName );
480  continue;
481  }
482  monitorElement->setSubscribed(subscribe);
483  dqm_info( "Monitor element: path {0}, name {1} has been {2}subscribed", path, eltName, subscribe ? "" : "un-" );
484  }
485  }
