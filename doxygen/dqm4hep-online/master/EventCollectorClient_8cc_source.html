83  void EventCollectorClient::setUpdateMode(const std::string &source, bool receiveUpdates) {
84  auto findIter = m_sourceInfoMap.find(source);
85  if(m_sourceInfoMap.end() == findIter) {
86  dqm_error( "EventCollectorClient::setUpdateMode: no source with '{0}' registered !", source );
87  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
88  }
89  bool subscribed = m_client.hasSubscribed(OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source));
90 
91  // two cases:
92  // 1) already subscribed and don't want to receive updates anymore
93  // 2) not subscribed and want to receive updates
94  if(subscribed != receiveUpdates) {
95  if(receiveUpdates) {
96  m_client.subscribe(
97  OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source),
98  &findIter->second,
99  &EventCollectorClient::SourceInfo::receiveEvent
100  );
101  }
102  else {
103  m_client.unsubscribe(
104  OnlineRoutes::EventCollector::eventUpdate(m_collectorName, source),
105  &findIter->second
106  );
107  }
108  }
109  }

143  core::EventPtr EventCollectorClient::readEvent(const net::Buffer &buffer) {
144  std::lock_guard<std::recursive_mutex> lock(m_mutex);
145 
146  try {
147  if(not m_bufferDevice) {
148  m_bufferDevice = std::make_shared<xdrstream::BufferDevice>((char*)buffer.begin(), buffer.size(), false);
149  m_bufferDevice->setOwner(false);
150  }
151  else {
152  m_bufferDevice->setBuffer((char*)buffer.begin(), buffer.size(), false);
153  }
154  }
155  catch(...) {
156  dqm_error( "EventCollectorClient::readEvent: couldn't setup buffer device !" );
157  return nullptr;
158  }
159  // read event using event streamer
160  core::EventPtr event(nullptr);
161  core::StatusCode statusCode = m_eventStreamer.readEvent(event, m_bufferDevice.get());
162 
163  if(core::STATUS_CODE_SUCCESS != statusCode) {
164  dqm_error( "EventCollectorClient::readEvent: streamer couldn't read event: {0}", core::statusCodeToString(statusCode) );
165  return nullptr;
166  }
167  return event;
168  }
