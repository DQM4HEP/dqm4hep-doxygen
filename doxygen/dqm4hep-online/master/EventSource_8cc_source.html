130  void EventSource::sendEvent(const core::StringVector &collectors, core::EventPtr event) {
131  if(not m_started) {
132  throw core::StatusCodeException(core::STATUS_CODE_NOT_INITIALIZED);
133  }
134 
135  if(nullptr == event) {
136  throw core::StatusCodeException(core::STATUS_CODE_INVALID_PTR);
137  }
138 
139  (void)m_bufferDevice->reset();
140  THROW_RESULT_IF(core::STATUS_CODE_SUCCESS, !=, m_eventStreamer.writeEvent(event, m_bufferDevice.get()));
141 
142  core::json sourceInfo;
143  net::Buffer collectBuffer;
144  auto model = collectBuffer.createModel();
145  collectBuffer.setModel(model);
146  model->handle(m_bufferDevice->getBuffer(), m_bufferDevice->getPosition());
147 
148  // send serialized event to all collectors
149  for(auto collector : collectors) {
150  auto iter = m_collectorInfos.find(collector);
151 
152  if(m_collectorInfos.end() == iter) {
153  dqm_error( "EventSource::sendEvent(): Collector '{0}' not found... Something is wrong !!!", collector );
154  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
155  }
156 
157  // Is the event collector server actually running ?
158  const std::string serverName(OnlineRoutes::Application::serverName(OnlineRoutes::EventCollector::applicationType(), iter->first));
159  if(!net::Server::isServerRunning(serverName)) {
160  // in case the server was stopped, we will need to send
161  // back the source registration on wake up
162  iter->second.m_registered = false;
163  continue;
164  }
165 
166  // if not yet registered to the collector, do it
167  if(!iter->second.m_registered) {
168  // get it once
169  if(sourceInfo.empty()) {
170  this->getSourceInfo(sourceInfo);
171  }
172  iter->second.m_registered = this->registerMe(iter->first, sourceInfo);
173  }
174 
175  if(!iter->second.m_registered) {
176  dqm_warning( "EventSource::sendEvent(): Source '{0}' not registered yet to collector '{1}'. Skipping ...", m_sourceName, collector );
177  continue;
178  }
179 
180  m_client.sendCommand(OnlineRoutes::EventCollector::collectEvent(collector), collectBuffer);
181  }
182  }

205  bool EventSource::registerMe(const std::string &collector, const core::json &info) {
206  std::string requestName = OnlineRoutes::EventCollector::registerSource(collector);
207  bool returnValue(false);
208  net::Buffer requestBuffer;
209  auto model = requestBuffer.createModel<std::string>();
210  requestBuffer.setModel(model);
211  std::string jsonDump(info.dump());
212  model->move(std::move(jsonDump));
213 
214  dqm_debug( "Sending request to collector {0} for registration, request: {1}", collector , requestName);
215  m_client.sendRequest(requestName, requestBuffer, [&returnValue,&collector](const net::Buffer &buffer){
216  core::json response({});
217 
218  if(0 != buffer.size()) {
219  response = core::json::parse(buffer.begin(), buffer.end());
220  }
221  const bool registered(response.value<bool>("registered", false));
222 
223  if(not registered) {
224  const std::string message(response.value<std::string>("message", "The event collector is maybe not available or is down !"));
225  dqm_warning( "EventSource::registerMe(): Couldn't register source to collector '{0}': {1}", collector, message );
226  returnValue = false;
227  }
228  else {
229  dqm_info( "Event source registered to event collector '{0}' !", collector );
230  returnValue = true;
231  }
232  });
233 
234  if(returnValue) {
235  dqm_info( "Will notify server on exit" );
236  const std::string serverName(OnlineRoutes::Application::serverName(OnlineRoutes::EventCollector::applicationType(), collector));
237  m_client.notifyServerOnExit(serverName);
238  }
239  return returnValue;
240  }
