290  inline void AppEventLoop::createTimer(const std::string &name, unsigned int nSeconds, bool singleShot,
291  Controller *controller, void (Controller::*function)()) {
292  std::lock_guard<std::recursive_mutex> lock(m_eventMutex);
293  auto findIter = m_timers.find(name);
294 
295  if(m_timers.end() != findIter) {
296  dqm_error( "AppEventLoop::createTimer: Timer with name '{0}' already exists", name );
297  throw core::StatusCodeException(core::STATUS_CODE_ALREADY_PRESENT);
298  }
299  // create a new timer
300  Timer *timer = new Timer(name, *this);
301  timer->setSingleShot(singleShot);
302  timer->setTimeout(nSeconds);
303  timer->onTimeout().connect(controller, function);
304  m_timers[name] = timer;
305 
306  if(running()) {
307  timer->startTimer();
308  }
309  }
