517  template <typename T>
518  inline void Application::sendStat(const std::string &entryName, const T &stats){
519  if(not statsEnabled() or noServer()) {
520  return;
521  }
522  core::json object = m_statistics[entryName];
523  if(object.is_null())
524  throw core::StatusCodeException(core::STATUS_CODE_NOT_FOUND);
525  // add metadata on the fly
526  object["name"] = entryName;
527  object["value"] = stats;
528  object["appType"] = this->type();
529  object["appName"] = this->name();
530  object["time"] = core::TimePoint::clock::to_time_t(core::now());
531  dqm_debug( "Sending app stat : \n'{0}'", object.dump() );
532  m_client.sendCommand(OnlineRoutes::OnlineManager::appStats(), object.dump());
533  }

574  auto findIter = m_serviceHandlerPtrMap.find(serviceName);

576  if(m_serviceHandlerPtrMap.end() != findIter) {
