227 
228  if( ! m_pInBuffer )
229  m_pInBuffer = new xdrstream::BufferDevice(pBuffer, bufferSize, false);

232 
233  m_pInBuffer->setOwner(false);
234 
235  StringVector subscribedMeList;
236 
237  if( ! XDR_TESTBIT( StreamingHelper::read( m_pInBuffer , subscribedMeList ) , xdrstream::XDR_SUCCESS ) )
238  {
239  LOG4CXX_ERROR( dqmMainLogger , "Couldn't read subscribed list" );
240  return;
241  }
242 
243  // update the list
244  pthread_mutex_lock(&m_mutex);
245  LOG4CXX_INFO( dqmMainLogger , "Received subscribed list size : " << subscribedMeList.size() );
246  m_subscribedMeList.clear();
247  m_subscribedMeList.insert( subscribedMeList.begin(), subscribedMeList.end() );
248  pthread_mutex_unlock(&m_mutex);
249  }
250  else if(pInfo == m_pCollectorStateInfo)
251  {
252  if(m_pCollectorStateInfo->getInt() == RUNNING_STATE)
253  {
254  LOG4CXX_INFO( dqmMainLogger , "Force update available list on next EOC !" );
255 
256  pthread_mutex_lock(&m_mutex);
257  m_sendAvailableMeList = true;
258  pthread_mutex_unlock(&m_mutex);
259  }

261  {
262  LOG4CXX_WARN( dqmMainLogger , "Server is down -> Clearing subscribed me list !" );
263 
264  pthread_mutex_lock(&m_mutex);

319  void MonitorElementSender::sendAvailableMonitorElementList()
320  {

322  return;

324  MonitorElementInfoList monitorElementInfoList;
325 
326  for(MonitorElementInfoMap::iterator iter = m_availableMeMap.begin(), endIter = m_availableMeMap.end() ;
327  endIter != iter ; ++iter)

329 
330  try

335  std::string moduleName = m_pApplication->getModule()->getName();
336
