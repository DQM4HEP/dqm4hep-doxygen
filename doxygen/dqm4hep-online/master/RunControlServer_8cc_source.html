96  {
97  m_stopFlag = false;

99  // create user external interface (plugin)
100  m_pInterface = PluginManager::instance()->create<RunControlInterface>(m_interfaceName);
101 
102  if( ! m_pInterface )

104  delete m_pServer;
105  dqm_error( "Couldn't find run control interface '{0}' in plugin manager", m_interfaceName );
106  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
107  }
108 
109  m_pInterface->setServer(this);
110  m_pInterface->readSettings(m_userParameters);
111 
112  // create network interface
113  m_pServer = new dqm4hep::net::Server(m_runControl.name());
114 
115  m_pSorService = m_pServer->createService(OnlineRoutes::RunControl::sor(m_runControl.name()));
116  m_runControl.onStartOfRun().connect(this, &RunControlServer::sor);
117 
118  m_pEorService = m_pServer->createService(OnlineRoutes::RunControl::eor(m_runControl.name()));
119  m_runControl.onEndOfRun().connect(this, &RunControlServer::eor);
120 
121  m_pServer->createRequestHandler(OnlineRoutes::RunControl::status(m_runControl.name()), this, &RunControlServer::sendCurrentRun);
122 
123  // start the server
124  m_pServer->start();
125 
126  // run the user plugin
127  if(m_pInterface->runBlocking())
128  {
129  // block here until stop called from outside
130  m_pInterface->run();
131  }
132  else
133  {
134  // call run and sleep
135  m_pInterface->run();
136 
137  while( ! m_stopFlag )
138  dqm4hep::core::sleep(dqm4hep::core::TimeDuration(1));
139  }
140 
141  // exit the server : stop and release memory
142  m_pServer->stop();
143 
144  delete m_pServer; m_pServer = nullptr;
145  delete m_pInterface; m_pInterface = nullptr;
146  m_pSorService = nullptr;
147  m_pEorService = nullptr;
148  }
