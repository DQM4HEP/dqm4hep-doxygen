72  void RunControlServer::run() {
73  m_stopFlag = false;
74 
75  // create user external interface (plugin)
76  m_interface = PluginManager::instance()->create<RunControlInterface>(m_interfaceName);
77 
78  if( not m_interface ) {
79  dqm_error( "Couldn't find run control interface '{0}' in plugin manager", m_interfaceName );
80  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
81  }
82 
83  // create network interface
84  m_server = std::make_shared<dqm4hep::net::Server>(m_runControl.name());
85 
86  m_pSorService = m_server->createService(OnlineRoutes::RunControl::sor(m_runControl.name()));
87  m_runControl.onStartOfRun().connect(this, &RunControlServer::sor);
88 
89  m_pEorService = m_server->createService(OnlineRoutes::RunControl::eor(m_runControl.name()));
90  m_runControl.onEndOfRun().connect(this, &RunControlServer::eor);
91 
92  m_server->createRequestHandler(OnlineRoutes::RunControl::status(m_runControl.name()), this, &RunControlServer::sendCurrentRun);
93 
94  m_interface->setServer(this);
95  m_interface->readSettings(m_userParameters);
96 
97  // start the server
98  m_server->start();
99 
100  // run the user plugin
101  if(m_interface->runBlocking())
102  {
103  // block here until stop called from outside
104  m_interface->run();
105  }
106  else
107  {
108  // call run and sleep
109  m_interface->run();
110 
111  while( ! m_stopFlag )
112  dqm4hep::core::sleep(dqm4hep::core::TimeDuration(1));
113  }
114 
115  m_server->stop();
116  m_pSorService = nullptr;
117  m_pEorService = nullptr;
118  }
