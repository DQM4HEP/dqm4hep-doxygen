113  void HttpRunControl::receiveStartOfRunPost(Mongoose::Request &request, Mongoose::StreamResponse &response) {
114  dqm_info("Received POST: /{0}/SOR", this->runControl().name() );
115 
116  if(runControl().isRunning()) {
117  response << "ERROR: Couldn't start new run, already running !" << std::endl;
118  return;
119  }
120 
121  std::map<std::string, std::string> variables(request.getAllVariable());
122  std::map<std::string, std::string> urlVariables;
123  request.getVars(urlVariables);
124 
125  dqm4hep::core::Run newRun;
126  std::string password;
127  bool consistent = false;
128 
129  for(auto iter = variables.begin(), endIter = variables.end() ; endIter != iter ; ++iter) {
130  if(iter->first == "run") {
131  int runNumber = 0;
132  dqm4hep::core::stringToType(iter->second, runNumber);
133  newRun.setRunNumber(runNumber);
134  consistent = true;
135  continue;
136  }
137  if(iter->first == "description") {
138  newRun.setDescription(iter->second);
139  continue;
140  }
141  if(iter->first == "detectorName") {
142  newRun.setDetectorName(iter->second);
143  continue;
144  }
145  if(iter->first == "password") {
146  password = iter->second;
147  continue;
148  }
149  newRun.setParameter(iter->first, iter->second);
150  }
151  if(not consistent) {
152  response << "ERROR: Run number required to start a new run !!"<< std::endl;
153  return;
154  }
155  try {
156  this->startNewRun(newRun, password);
157  dqm_info( "*** Started new run ***" );
158  dqm_info( "{0}", core::typeToString(newRun) );
159  dqm_info( "***********************" );
160  }
161  catch( dqm4hep::core::StatusCodeException &exception ) {
162  response << "ERROR: Couldn't start new run ! What :" << exception.toString()<< std::endl;
163  return;
164  }
165  response << "SUCCESS\n";
166  }

170  void HttpRunControl::receiveEndOfRunPost(Mongoose::Request &request, Mongoose::StreamResponse &response) {
171  dqm_info("Received POST: /{0}/EOR", this->runControl().name());
172  if(! this->runControl().isRunning()) {
173  response << "ERROR: Not running ! Couldn't end current run !";
174  return;
175  }
176  std::map<std::string, std::string> variables(request.getAllVariable());
177  std::string password;
178  dqm4hep::core::StringMap parameters;
179 
180  for(auto iter = variables.begin(), endIter = variables.end() ; endIter != iter ; ++iter) {
181  if( iter->first == "password" ) {
182  password = iter->second;
183  continue;
184  }
185  parameters[iter->first] = iter->second;
186  }
187 
188  try {
189  this->endCurrentRun(parameters, password);
190  }
191  catch( dqm4hep::core::StatusCodeException &exception ) {
192  response << "ERROR: Couldn't stop current run ! Exception: " << exception.toString() << std::endl;
193  return;
194  }
195 
196  response << "SUCCESS";
197  }
