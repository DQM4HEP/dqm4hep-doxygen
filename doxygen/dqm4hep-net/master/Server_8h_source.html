338  void (Controller::*function)(const Buffer &request, Buffer &response)) {
339  auto findIter = m_requestHandlerMap.find(name);
340 
341  if (findIter != m_requestHandlerMap.end())
342  throw std::runtime_error("Server::createRequestHandler(): request handler '" + name +
343  "' already exists in this client");
344 
345  if (Server::requestHandlerAlreadyRunning(name))
346  throw std::runtime_error("Server::createRequestHandler(): request handler '" + name +
347  "' already running on network");
348 
349  // first insert nullptr, then create request handler
350  std::pair<RequestHandlerMap::iterator, bool> inserted =
351  m_requestHandlerMap.insert(RequestHandlerMap::value_type(name, nullptr));
352 
353  if (inserted.second) {
354  RequestHandler *pRequestHandler = new RequestHandler(this, name, pController, function);
355  inserted.first->second = pRequestHandler;
