778  void read_message(const std::shared_ptr<Connection> &connection, const std::shared_ptr<boost::asio::streambuf> &read_buffer, Endpoint& endpoint) const {
779  boost::asio::async_read(*connection->socket, *read_buffer, boost::asio::transfer_exactly(2),
780  [this, connection, read_buffer, &endpoint]
781  (const boost::system::error_code& ec, size_t bytes_transferred) {
782  if(!ec) {
783  if(bytes_transferred==0) { //TODO: why does this happen sometimes?
784  read_message(connection, read_buffer, endpoint);
785  return;
786  }
787  std::istream stream(read_buffer.get());
788 
789  std::vector<unsigned char> first_bytes;
790  first_bytes.resize(2);
791  stream.read((char*)&first_bytes[0], 2);
792 
793  unsigned char fin_rsv_opcode=first_bytes[0];
794 
795  //Close connection if unmasked message from client (protocol error)
796  if(first_bytes[1]<128) {
797  const std::string reason("message from client not masked");
798  send_close(connection, 1002, reason, [this, connection](const boost::system::error_code& /*ec*/) {});
799  connection_close(connection, endpoint, 1002, reason);
800  return;
801  }
802 
803  size_t length=(first_bytes[1]&127);
804 
805  if(length==126) {
806  //2 next bytes is the size of content
807  boost::asio::async_read(*connection->socket, *read_buffer, boost::asio::transfer_exactly(2),
808  [this, connection, read_buffer, &endpoint, fin_rsv_opcode]
809  (const boost::system::error_code& ec, size_t /*bytes_transferred*/) {
810  if(!ec) {
811  std::istream stream(read_buffer.get());
812 
813  std::vector<unsigned char> length_bytes;
814  length_bytes.resize(2);
815  stream.read((char*)&length_bytes[0], 2);
816 
817  size_t length=0;
818  int num_bytes=2;
819  for(int c=0;c<num_bytes;c++)
820  length+=length_bytes[c]<<(8*(num_bytes-1-c));
821 
822  read_message_content(connection, read_buffer, length, endpoint, fin_rsv_opcode);
823  }
824  else
825  connection_error(connection, endpoint, ec);
826  });
827  }
828  else if(length==127) {
829  //8 next bytes is the size of content
830  boost::asio::async_read(*connection->socket, *read_buffer, boost::asio::transfer_exactly(8),
831  [this, connection, read_buffer, &endpoint, fin_rsv_opcode]
832  (const boost::system::error_code& ec, size_t /*bytes_transferred*/) {
833  if(!ec) {
834  std::istream stream(read_buffer.get());
835 
836  std::vector<unsigned char> length_bytes;
837  length_bytes.resize(8);
838  stream.read((char*)&length_bytes[0], 8);
839 
840  size_t length=0;
841  int num_bytes=8;
842  for(int c=0;c<num_bytes;c++)
843  length+=length_bytes[c]<<(8*(num_bytes-1-c));
844 
845  read_message_content(connection, read_buffer, length, endpoint, fin_rsv_opcode);
846  }
847  else
848  connection_error(connection, endpoint, ec);
849  });
850  }
851  else
852  read_message_content(connection, read_buffer, length, endpoint, fin_rsv_opcode);
853  }
854  else
855  connection_error(connection, endpoint, ec);
856  });
857  }

859  void read_message_content(const std::shared_ptr<Connection> &connection, const std::shared_ptr<boost::asio::streambuf> &read_buffer, size_t length, Endpoint& endpoint, unsigned char fin_rsv_opcode) const {
860  boost::asio::async_read(*connection->socket, *read_buffer, boost::asio::transfer_exactly(4+length),
861  [this, connection, read_buffer, length, &endpoint, fin_rsv_opcode]
862  (const boost::system::error_code& ec, size_t /*bytes_transferred*/) {
863  if(!ec) {
864  std::istream raw_message_data(read_buffer.get());
865 
866  //Read mask
867  std::vector<unsigned char> mask;
868  mask.resize(4);
869  raw_message_data.read((char*)&mask[0], 4);
870 
871  std::shared_ptr<Message> message(new Message());
872  message->length=length;
873  message->fin_rsv_opcode=fin_rsv_opcode;
874 
875  std::ostream message_data_out_stream(&message->streambuf);
876  for(size_t c=0;c<length;c++) {
877  message_data_out_stream.put(raw_message_data.get()^mask[c%4]);
878  }
879 
880  //If connection close
881  if((fin_rsv_opcode&0x0f)==8) {
882  int status=0;
883  if(length>=2) {
884  unsigned char byte1=message->get();
885  unsigned char byte2=message->get();
886  status=(byte1<<8)+byte2;
887  }
888 
889  auto reason=message->string();
890  send_close(connection, status, reason, [this, connection](const boost::system::error_code& /*ec*/) {});
891  connection_close(connection, endpoint, status, reason);
892  return;
893  }
894  else {
895  //If ping
896  if((fin_rsv_opcode&0x0f)==9) {
897  //send pong
898  auto empty_send_stream=std::make_shared<SendStream>();
899  send(connection, empty_send_stream, nullptr, fin_rsv_opcode+1);
900  }
901  else if(endpoint.on_message) {
902  timer_idle_reset(connection);
903  endpoint.on_message(connection, message);
904  }
905 
906  //Next message
907  read_message(connection, read_buffer, endpoint);
908  }
909  }
910  else
911  connection_error(connection, endpoint, ec);
912  });
913  }
