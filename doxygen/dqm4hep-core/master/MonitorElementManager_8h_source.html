319  template <typename ObjectType, typename... Args>
320  StatusCode MonitorElementManager::bookObject(const std::string &path, const std::string &name,
321  MonitorElementPtr &monitorElement,
322  allocator_helper<TObject, ObjectType, Args...> allocator,
323  Args... args) {
324  monitorElement = nullptr;
325 
326  if (!ObjectType::Class()->HasDictionary()) {
327  dqm_error("Couldn't book object of type '{0}', because this class has no dictionnary!",
328  ObjectType::Class_Name());
329  return STATUS_CODE_NOT_ALLOWED;
330  }
331 
332  if (ObjectType::Class()->IsForeign()) {
333  dqm_error(
334  "Couldn't book object of type '{0}', because this class is foreign (does not have a Streamer method))!",
335  ObjectType::Class_Name());
336  return STATUS_CODE_NOT_ALLOWED;
337  }
338 
339  if (!ObjectType::Class()->InheritsFrom("TNamed")) {
340  dqm_error("Couldn't book object of type '{0}', because this class doesn't inherit TNamed (required to call "
341  "SetName())!",
342  ObjectType::Class_Name());
343  return STATUS_CODE_NOT_ALLOWED;
344  }
345 
346  const bool objectStat(TObject::GetObjectStat());
347  TObject::SetObjectStat(false);
348  TObject *pTObject = allocator.create(args...);
349  TObject::SetObjectStat(objectStat);
350 
351  if (!pTObject) {
352  dqm_warning("Couldn't allocate monitor element of type '{0}', path '{1}', name '{2}'", ObjectType::Class_Name(),
353  path, name);
354  return STATUS_CODE_FAILURE;
355  }
356 
357  ((TNamed *)pTObject)->SetName(name.c_str());
358 
359  return this->addMonitorElement(path, pTObject, monitorElement);
360  }
361  }
362 }
