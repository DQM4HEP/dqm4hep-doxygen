560  template <typename T>
561  inline StatusCode MonitorElementManager::readMonitorElements(TiXmlElement *const pXmlElement, bool readQTests) {
562  if(nullptr == pXmlElement) {
563  dqm_error( "MonitorElementManager::readMonitorElements: Invalid xml element !" );
564  return STATUS_CODE_INVALID_PTR;
565  }
566  for (TiXmlElement *child = pXmlElement->FirstChildElement(); child != nullptr; child = child->NextSiblingElement()) {
567  std::shared_ptr<T> monitorElement;
568  if(child->ValueStr() == "bookElement") {
569  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, bookMonitorElement<T>(child, monitorElement));
570  }
571  else if(child->ValueStr() == "fileElement") {
572  std::string rootFile;
573  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(child, "file", rootFile));
574  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, readMonitorElement<T>(rootFile, child, monitorElement));
575  }
576  else if(child->ValueStr() == "file") {
577  std::string rootFile;
578  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(child, "name", rootFile));
579  for (TiXmlElement *child2 = child->FirstChildElement("fileElement"); child2 != nullptr; child2 = child2->NextSiblingElement("fileElement")) {
580  // read sub-element
581  std::shared_ptr<T> monitorElement2;
582  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, readMonitorElement<T>(rootFile, child2, monitorElement2));
583  // attach quality tests if option set
584  if(readQTests) {
585  for (TiXmlElement *qtest = child2->FirstChildElement("qtest"); qtest != nullptr; qtest = qtest->NextSiblingElement("qtest")) {
586  std::string qTestName;
587  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(qtest, "name", qTestName));
588  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, addQualityTest(monitorElement2->path(), monitorElement2->name(), qTestName));
589  }
590  }
591  }
592  continue;
593  }
594  else {
595  continue;
596  }
597  // attach quality tests if option set
598  if(readQTests) {
599  for (TiXmlElement *qtest = child->FirstChildElement("qtest"); qtest != nullptr; qtest = qtest->NextSiblingElement("qtest")) {
600  std::string qTestName;
601  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(qtest, "name", qTestName));
602  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, addQualityTest(monitorElement->path(), monitorElement->name(), qTestName));
603  }
604  }
605  }
606  return STATUS_CODE_SUCCESS;
607  }
