322  template <typename ObjectType, typename... Args>
323  StatusCode MonitorElementManager::bookObject(const std::string &path, const std::string &name,
324  MonitorElementPtr &monitorElement,
325  allocator_helper<TObject, ObjectType, Args...> allocator,
326  Args... args) {
327  monitorElement = nullptr;
328 
329  if (!ObjectType::Class()->HasDictionary()) {
330  dqm_error("Couldn't book object of type '{0}', because this class has no dictionnary!",
331  ObjectType::Class_name());
332  return STATUS_CODE_NOT_ALLOWED;
333  }
334 
335  if (ObjectType::Class()->IsForeign()) {
336  dqm_error(
337  "Couldn't book object of type '{0}', because this class is foreign (does not have a Streamer method))!",
338  ObjectType::Class_name());
339  return STATUS_CODE_NOT_ALLOWED;
340  }
341 
342  if (!ObjectType::Class()->InheritsFrom("TNamed")) {
343  dqm_error("Couldn't book object of type '{0}', because this class doesn't inherit TNamed (required to call "
344  "SetName())!",
345  ObjectType::Class_name());
346  return STATUS_CODE_NOT_ALLOWED;
347  }
348 
349  const bool objectStat(TObject::GetObjectStat());
350  TObject::SetObjectStat(false);
351  TObject *pTObject = allocator.create(args...);
352  TObject::SetObjectStat(objectStat);
353 
354  if (!pTObject) {
355  dqm_warning("Couldn't allocate monitor element of type '{0}', path '{1}', name '{2}'", ObjectType::Class_name(),
356  path, name);
357  return STATUS_CODE_FAILURE;
358  }
359 
360  ((TNamed *)pTObject)->SetName(name.c_str());
361 
362  return this->addMonitorElement(path, pTObject, monitorElement);
363  }
364  }
365 }
