125  auto pos = device->getPosition();
126  unsigned int nWritten = 0;
127  XDRSTREAM_SUCCESS_RESTORE(device->write(&nWritten), pos);
128  for(auto element : elements) {
129  auto eltPos = device->getPosition();
130  StatusCode writeCode = writeMonitorElement(element, device);
131  if(writeCode != STATUS_CODE_SUCCESS) {
132  if(returnOnFirstFail) {
133  // seek at begining and return
134  device->seek(pos);
135  return writeCode;
136  }
137  else {
138  // seek before current element
139  device->seek(eltPos);
140  }
141  }
142  nWritten++;
143  }
144  // write the actual number of written monitor element at start position
145  auto endPos = device->getPosition();
146  device->seek(pos);
147  XDRSTREAM_SUCCESS_RESTORE(device->write(&nWritten), pos);
148  device->seek(endPos);
