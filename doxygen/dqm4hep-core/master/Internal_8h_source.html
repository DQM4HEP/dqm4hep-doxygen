393  inline void tokenize(const std::string &inputString, std::vector<T> &tokens, const std::string &delimiter = " ") {
394  std::string::size_type lastPos = inputString.find_first_not_of(delimiter, 0);
395  std::string::size_type pos = inputString.find_first_of(delimiter, lastPos);
396 
397  while ((std::string::npos != pos) || (std::string::npos != lastPos)) {
398  T value;
399  if(not stringToType(inputString.substr(lastPos, pos - lastPos), value)) {
400  return;
401  }
402  tokens.push_back(value);
403  lastPos = inputString.find_first_not_of(delimiter, pos);
404  pos = inputString.find_first_of(delimiter, lastPos);
405  }
406  }
