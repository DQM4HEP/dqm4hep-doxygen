230  inline void tokenize(const std::string &inputString, StringVector &tokens, const std::string &delimiter = " ") {
231  std::string::size_type lastPos = inputString.find_first_not_of(delimiter, 0);
232  std::string::size_type pos = inputString.find_first_of(delimiter, lastPos);
233 
234  while ((std::string::npos != pos) || (std::string::npos != lastPos)) {
235  tokens.push_back(inputString.substr(lastPos, pos - lastPos));
236  lastPos = inputString.find_first_not_of(delimiter, pos);
237  pos = inputString.find_first_of(delimiter, lastPos);
238  }
239  }
