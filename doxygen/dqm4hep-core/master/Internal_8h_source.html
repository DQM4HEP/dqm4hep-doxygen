234  inline void tokenize(const std::string &inputString, std::vector<T> &tokens, const std::string &delimiter = " ") {
235  std::string::size_type lastPos = inputString.find_first_not_of(delimiter, 0);
236  std::string::size_type pos = inputString.find_first_of(delimiter, lastPos);
237 
238  while ((std::string::npos != pos) || (std::string::npos != lastPos)) {
239  T value;
240  if(not stringToType(inputString.substr(lastPos, pos - lastPos), value)) {
241  return;
242  }
243  tokens.push_back(value);
244  lastPos = inputString.find_first_not_of(delimiter, pos);
245  pos = inputString.find_first_of(delimiter, lastPos);
246  }
247  }
