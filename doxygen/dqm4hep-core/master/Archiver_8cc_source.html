58  // if already open write the archive if not done
59  // and close it before to re-open
60  if (m_isOpened)

69  size_t pos = archiveFileName.rfind(".root");
70

170  StatusCode Archiver::recursiveFill(MonitorElementDir directory, TDirectory *pROOTDir) {
171  if (nullptr == directory || nullptr == pROOTDir)
172  return STATUS_CODE_INVALID_PTR;
173 
174  pROOTDir->cd();
175 
176  const auto &subDirList(directory->subdirs());
177 
178  if (!subDirList.empty()) {
179  for (auto iter = subDirList.begin(), endIter = subDirList.end(); endIter != iter; ++iter) {
180  auto subDir = *iter;
181  TDirectory *pROOTSubDir = pROOTDir->mkdir(subDir->name().c_str());
182 
183  if (nullptr != pROOTSubDir) {
184  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, Archiver::recursiveFill(subDir, pROOTSubDir));
185  }
186  }
187  }
188 
189  // write the monitor elements

191  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, Archiver::writeMonitorElements(directory, pROOTDir));
