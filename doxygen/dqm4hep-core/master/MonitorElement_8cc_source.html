375  void TDynamicGraph::ShrinkToRange() {
376  if (!TestBit(kDynamicRange) || !TestBit(kShrinkToRange))
377  return;
378 
379  Int_t nPoints = this->GetN();
380 
381  if (m_rangeLength <= 0 || nPoints == 0)
382  return;
383 
384  Double_t lastX, lastY;
385  Int_t res = this->GetPoint(nPoints - 1, lastX, lastY);
386
