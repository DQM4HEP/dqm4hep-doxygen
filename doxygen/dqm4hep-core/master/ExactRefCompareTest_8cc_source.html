169  void ExactRefCompareTest::doHistogramTest(MonitorElement* monitorElement, QualityTestReport &report) {
170 
171  TH1 *histogram = monitorElement->objectTo<TH1>();
172  TH1 *reference = monitorElement->referenceTo<TH1>();
173  int nBins(0), nBinsRef(0);
174 
175  if(histogram->GetDimension() == 1) {
176  nBins = histogram->GetNbinsX();
177  nBinsRef = reference->GetNbinsX();
178  }
179  else if(histogram->GetDimension() == 2) {
180  nBins = histogram->GetNbinsX() * histogram->GetNbinsY();
181  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY();
182  }
183  else if(histogram->GetDimension() == 3) {
184  nBins = histogram->GetNbinsX() * histogram->GetNbinsY() * histogram->GetNbinsZ();
185  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY() * reference->GetNbinsZ();
186  }
187  else {
188  report.m_message = "Histogram with invalid dimension (dim = " + typeToString(histogram->GetDimension()) + ") !";
189  report.m_quality = 0.f;
190  throw StatusCodeException(STATUS_CODE_FAILURE);
191  }
192 
193  if(nBins != nBinsRef) {
194  report.m_message = "Histogram and reference do not have the same number of bins";
195  report.m_quality = 0.f;
196  throw StatusCodeException(STATUS_CODE_FAILURE);
197  }
198 
199  const int firstBin = m_compareUnderflow ? 0 : 1;
200  const int lastBin = m_compareOverflow ? nBins : nBins+1;
201  unsigned int nDifferentBins(0);
202 
203  for(auto bin = firstBin ; bin <= lastBin ; ++bin) {
204 
205  const Double_t binContent(histogram->GetBinContent(bin));
206  const Double_t binContentRef(reference->GetBinContent(bin));
207  const bool binsAreDifferent(!(fabs(binContent-binContentRef) < std::numeric_limits<Double_t>::epsilon()));
208 
209  if(binsAreDifferent) {
210  ++nDifferentBins;
211  }
212  }
213 
214  if(nDifferentBins != 0) {
215  report.m_message = "Histogram and reference are different (difference = " + typeToString(nDifferentBins) + ")";
216  // Quality estimate as the percentage of different points.
217  // This factor is scaled to the error limit to make sure the quality flag is set to ERROR
218  report.m_quality = this->errorLimit() * (static_cast<float>(nBins) - static_cast<float>(nDifferentBins)) / static_cast<float>(nBins);
219  }
220  else {
221  report.m_message = "Histogram and reference are equal !";
222  report.m_quality = 1.f;
223  }
224 
225  report.m_extraInfos["nDifferences"] = nDifferentBins;
226  }
