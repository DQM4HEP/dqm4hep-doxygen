171  void ExactRefCompareTest::doHistogramTest(MonitorElementPtr monitorElement, QualityTestReport &report) {
172 
173  TH1 *histogram = monitorElement->objectTo<TH1>();
174  TH1 *reference = monitorElement->referenceTo<TH1>();
175  int nBins(0), nBinsRef(0);
176 
177  if(histogram->GetDimension() == 1) {
178  nBins = histogram->GetNbinsX();
179  nBinsRef = reference->GetNbinsX();
180  }
181  else if(histogram->GetDimension() == 2) {
182  nBins = histogram->GetNbinsX() * histogram->GetNbinsY();
183  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY();
184  }
185  else if(histogram->GetDimension() == 3) {
186  nBins = histogram->GetNbinsX() * histogram->GetNbinsY() * histogram->GetNbinsZ();
187  nBinsRef = reference->GetNbinsX() * reference->GetNbinsY() * reference->GetNbinsZ();
188  }
189  else {
190  report.m_message = "Histogram with invalid dimension (dim = " + typeToString(histogram->GetDimension()) + ") !";
191  report.m_quality = 0.f;
192  throw StatusCodeException(STATUS_CODE_FAILURE);
193  }
194 
195  if(nBins != nBinsRef) {
196  report.m_message = "Histogram and reference do not have the same number of bins";
197  report.m_quality = 0.f;
198  throw StatusCodeException(STATUS_CODE_FAILURE);
199  }
200 
201  const int firstBin = m_compareUnderflow ? 0 : 1;
202  const int lastBin = m_compareOverflow ? nBins : nBins+1;
203  unsigned int nDifferentBins(0);
204 
205  for(auto bin = firstBin ; bin <= lastBin ; ++bin) {
206 
207  const Double_t binContent(histogram->GetBinContent(bin));
208  const Double_t binContentRef(reference->GetBinContent(bin));
209  const bool binsAreDifferent(!(fabs(binContent-binContentRef) < std::numeric_limits<Double_t>::epsilon()));
210 
211  if(binsAreDifferent) {
212  ++nDifferentBins;
213  }
214  }
215 
216  if(nDifferentBins != 0) {
217  report.m_message = "Histogram and reference are different (difference = " + typeToString(nDifferentBins) + ")";
218  // Quality estimate as the percentage of different points.
219  // This factor is scaled to the error limit to make sure the quality flag is set to ERROR
220  report.m_quality = this->errorLimit() * (static_cast<float>(nBins) - static_cast<float>(nDifferentBins)) / static_cast<float>(nBins);
221  }
222  else {
223  report.m_message = "Histogram and reference are equal !";
224  report.m_quality = 1.f;
225  }
226 
227  report.m_extraInfos["nDifferences"] = nDifferentBins;
228  }
