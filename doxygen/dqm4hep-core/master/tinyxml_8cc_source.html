55  void TiXmlBase::EncodeString(const TIXML_STRING &str, TIXML_STRING *outString) {
56  int i = 0;
57 
58  while (i < (int)str.length()) {
59  unsigned char c = (unsigned char)str[i];
60 
61  if (c == '&' && i < ((int)str.length() - 2) && str[i + 1] == '#' && str[i + 2] == 'x') {
62  // Hexadecimal character reference.
63  // Pass through unchanged.
64  // &#xA9; -- copyright symbol, for example.
65  //
66  // The -1 is a bug fix from Rob Laveaux. It keeps
67  // an overflow from happening if there is no ';'.
68  // There are actually 2 ways to exit this loop -
69  // while fails (error case) and break (semicolon found).
70  // However, there is no mechanism (currently) for
71  // this function to return an error.
72  while (i < (int)str.length() - 1) {
73  outString->append(str.c_str() + i, 1);
74  ++i;
75  if (str[i] == ';')
76  break;
77  }
78  } else if (c == '&') {
79  outString->append(entity[0].str, entity[0].strLength);
80  ++i;
81  } else if (c == '<') {
82  outString->append(entity[1].str, entity[1].strLength);
83  ++i;
84  } else if (c == '>') {
85  outString->append(entity[2].str, entity[2].strLength);
86  ++i;
87  } else if (c == '\"') {
88  outString->append(entity[3].str, entity[3].strLength);
89  ++i;
90  } else if (c == '\'') {
91  outString->append(entity[4].str, entity[4].strLength);
92  ++i;
93  } else if (c < 32) {
94  // Easy pass at non-alpha/numeric/symbol
95  // Below 32 is symbolic.
96  char buf[32];
97 
98 #if defined(TIXML_SNPRINTF)
99  TIXML_SNPRINTF(buf, sizeof(buf), "&#x%02X;", (unsigned)(c & 0xff));
100 #else
101  sprintf(buf, "&#x%02X;", (unsigned)(c & 0xff));
102 #endif
103 
104  //*ME: warning C4267: convert 'size_t' to 'int'
105  //*ME: Int-Cast to make compiler happy ...
106  outString->append(buf, (int)strlen(buf));
107  ++i;
108  } else {
109  // char realc = (char) c;
110  // outString->append( &realc, 1 );
111  *outString += (char)c; // somewhat more efficient function call.
112  ++i;
113  }
114  }

615  void TiXmlElement::Print(FILE *cfile, int depth) const {
616  int i;
617  assert(cfile);
618  for (i = 0; i < depth; i++) {
619  fprintf(cfile, " ");
620  }
621 
622  fprintf(cfile, "<%s", value.c_str());
623 
624  const TiXmlAttribute *attrib;
625  for (attrib = attributeSet.First(); attrib != nullptr; attrib = attrib->Next()) {
626  fprintf(cfile, " ");
627  attrib->Print(cfile, depth);
628  }
629 
630  // There are 3 different formatting approaches:
631  // 1) An element without children is printed as a <foo /> node
632  // 2) An element with only a text child is printed as <foo> text </foo>
633  // 3) An element with children is printed on multiple lines.
634  TiXmlNode *node;
635  if (firstChild == nullptr) {
636  fprintf(cfile, " />");
637  } else if (firstChild == lastChild && (firstChild->ToText() != nullptr)) {
638  fprintf(cfile, ">");
639  firstChild->Print(cfile, depth + 1);
640  fprintf(cfile, "</%s>", value.c_str());
641  } else {
642  fprintf(cfile, ">");
643 
644  for (node = firstChild; node != nullptr; node = node->NextSibling()) {
645  if (node->ToText() == nullptr) {
646  fprintf(cfile, "\n");
647  }
648  node->Print(cfile, depth + 1);
649  }
650  fprintf(cfile, "\n");
651  for (i = 0; i < depth; ++i) {
652  fprintf(cfile, " ");
653  }
654  fprintf(cfile, "</%s>", value.c_str());
655  }
656  }

762  bool TiXmlDocument::LoadFile(FILE *file, TiXmlEncoding encoding) {
763  if (file == nullptr) {
764  SetError(TIXML_ERROR_OPENING_FILE, nullptr, nullptr, TIXML_ENCODING_UNKNOWN);
765  return false;
766  }
767 
768  // Delete the existing data:
769  Clear();
770  location.Clear();
771 
772  // Get the file size, so we can pre-allocate the string. HUGE speed impact.
773  long length = 0;
774  fseek(file, 0, SEEK_END);
775  length = ftell(file);
776  fseek(file, 0, SEEK_SET);
777 
778  // Strange case, but good to handle up front.
779  if (length <= 0) {
780  SetError(TIXML_ERROR_DOCUMENT_EMPTY, nullptr, nullptr, TIXML_ENCODING_UNKNOWN);
781  return false;
782  }
783 
784  // Subtle bug here. TinyXml did use fgets. But from the XML spec:
785  // 2.11 End-of-Line Handling
786  // <snip>
787  // <quote>
788  // ...the XML processor MUST behave as if it normalized all line breaks in external
789  // parsed entities (including the document entity) on input, before parsing, by translating
790  // both the two-character sequence #xD #xA and any #xD that is not followed by #xA to
791  // a single #xA character.
792  // </quote>
793  //
794  // It is not clear fgets does that, and certainly isn't clear it works cross platform.
795  // Generally, you expect fgets to translate from the convention of the OS to the c/unix
796  // convention, and not work generally.
797 
798  /*
799  while( fgets( buf, sizeof(buf), file ) )
800  {
801  data += buf;
802  }
803  */
804 
805  auto buf = new char[length + 1];
806  buf[0] = 0;
807 
808  if (fread(buf, length, 1, file) != 1) {
809  delete[] buf;
810  SetError(TIXML_ERROR_OPENING_FILE, nullptr, nullptr, TIXML_ENCODING_UNKNOWN);
811  return false;
812  }
813 
814  // Process the buffer in place to normalize new lines. (See comment above.)
815  // Copies from the 'p' to 'q' pointer, where p can advance faster if
816  // a newline-carriage return is hit.
817  //
818  // Wikipedia:
819  // Systems based on ASCII or a compatible character set use either LF (Line feed, '\n', 0x0A, 10 in decimal) or
820  // CR (Carriage return, '\r', 0x0D, 13 in decimal) individually, or CR followed by LF (CR+LF, 0x0D 0x0A)...
821  // * LF: Multics, Unix and Unix-like systems (GNU/Linux, AIX, Xenix, Mac OS X, FreeBSD, etc.),
822  // BeOS, Amiga, RISC OS, and others
823  // * CR+LF: DEC RT-11 and most other early non-Unix, non-IBM OSes, CP/M, MP/M, DOS, OS/2, Microsoft
824  // Windows, Symbian OS
825  // * CR: Commodore 8-bit machines, Apple II family, Mac OS up to version 9 and OS-9
826 
827  const char *p = buf; // the read head
828  char *q = buf; // the write head
829  const char CR = 0x0d;
830  const char LF = 0x0a;
831 
832  buf[length] = 0;
833  while (*p != 0) {
834  assert(p < (buf + length));
835  assert(q <= (buf + length));
836  assert(q <= p);
837 
838  if (*p == CR) {
839  *q++ = LF;
840  p++;
841  if (*p == LF) { // check for CR+LF (and skip LF)
842  p++;
843  }
844  } else {
845  *q++ = *p++;
846  }
847  }
848  assert(q <= (buf + length));
849  *q = 0;
850 
851  Parse(buf, nullptr, encoding);
852 
853  delete[] buf;
854  return !Error();
855  }
