52  static void procCpuStats(ProcessStats &stats) {
53  struct rusage ru;
54  if (getrusage(RUSAGE_SELF, &ru) < 0) {
55  dqm_error("[{0}] - Failed to getrusage", __FUNCTION__);
56  throw core::StatusCodeException(STATUS_CODE_FAILURE);
57  } else {
58 
59  dqm_float prevCpuTimeUser = stats.cpuTimeUser;
60  dqm_float prevCpuTimeSys = stats.cpuTimeSys;
61  stats.cpuTimeUser = (dqm_float)(ru.ru_utime.tv_sec) + ((dqm_float)(ru.ru_utime.tv_usec) / 1000000.);
62  stats.cpuTimeSys = (dqm_float)(ru.ru_stime.tv_sec) + ((dqm_float)(ru.ru_stime.tv_usec) / 1000000.);
63 
64  struct timeval timeNow;
65  gettimeofday(&timeNow, NULL);
66  dqm_double start = stats.lastPollTime.tv_sec + stats.lastPollTime.tv_usec / 1000000;
67  dqm_double stop = timeNow.tv_sec + timeNow.tv_usec / 1000000;
68  stats.lastPollTime = timeNow;
69  if (stop > start) {
70  dqm_double timeDif = stop - start;
71  // Don't compute percentage for the first call
72  if (0 != prevCpuTimeUser || 0 != prevCpuTimeSys) {
73  stats.cpuUser = (stats.cpuTimeUser - prevCpuTimeUser) / timeDif;
74  stats.cpuSys = (stats.cpuTimeSys - prevCpuTimeSys) / timeDif;
75  stats.cpuTot = stats.cpuUser + stats.cpuSys;
76  stats.cpuUser *= 100.;
77  stats.cpuSys *= 100.;
78  stats.cpuTot *= 100.;
79  }
80  } else {
81  dqm_error("[{0}] - Something went wrong when comparing clocks", __FUNCTION__);
82  stats.cpuUser = -1;
83  stats.cpuSys = -1;
84  stats.cpuTot = -1;
85  }
86  }
87  }

445  static void darwinMemStats(MemoryStats &stats) {
446  vm_statistics_data_t vm_info;
447  mach_msg_type_number_t count = HOST_VM_INFO_COUNT;
448 
449  kern_return_t kr = host_statistics(mach_host_self(), HOST_VM_INFO, (host_info_t)&vm_info, &count);
450  if (kr != KERN_SUCCESS) {
451  dqm_error("[{0}] - Failed to get host_statistics: {1}", __FUNCTION__, mach_error_string(kr));
452  throw core::StatusCodeException(STATUS_CODE_FAILURE);
453  }
454  static dqm_int pshift = 0;
455  if (pshift == 0) {
456  for (int psize = getpagesize(); psize > 1; psize >>= 1) {
457  pshift++;
458  }
459  }
460 
461  dqm_long_long used = (dqm_long_long)(vm_info.active_count + vm_info.inactive_count + vm_info.wire_count) << pshift;
462  dqm_long_long free = (dqm_long_long)(vm_info.free_count) << pshift;
463  dqm_long_long total =
464  (dqm_long_long)(vm_info.active_count + vm_info.inactive_count + vm_info.free_count + vm_info.wire_count)
465  << pshift;
466 
467  // Swap is available at same time as mem, so grab values here.
468  dqm_long_long swap_used = vm_info.pageouts << pshift;
469 
470  // Figure out total swap. This adds up the size of the swapfiles */
471  static DIR *dirp = nullptr;
472  dirp = opendir("/private/var/vm");
473  if (!dirp) {
474  dqm_error("[{0}] - Failed to get vm file", __FUNCTION__);
475  throw core::StatusCodeException(STATUS_CODE_FAILURE);
476  // return;
477  }
478 
479  dqm_long_long swap_total = 0;
480  struct dirent *dp = nullptr;
481 
482  while ((dp = readdir(dirp)) != 0) {
483  struct stat sb;
484  char fname[MAXNAMLEN];
485  if (strncmp(dp->d_name, "swapfile", 8))
486  continue;
487  strlcpy(fname, "/private/var/vm/", MAXNAMLEN);
488  strlcat(fname, dp->d_name, MAXNAMLEN);
489  if (stat(fname, &sb) < 0)
490  continue;
491  swap_total += sb.st_size;
492  }
493  closedir(dirp);
494 
495  stats.vmTot = (dqm_int)((total + swap_total) >> 20); // divide by 1024 * 1024
496  stats.vmUsed = (dqm_int)((used + swap_used) >> 20);
497  stats.vmFree = (dqm_int)((free + swap_total - swap_used) >> 20);
498  stats.rssTot = (dqm_int)(total >> 20);
499  stats.rssUsed = (dqm_int)(used >> 20);
500  }

508  static void darwinProcStats(ProcessStats &stats) {
509 #ifdef _LP64
510 #define vm_region vm_region_64
511 #endif
512 
513 // taken from <mach/shared_memory_server.h> which is obsoleted in 10.5
514 #define GLOBAL_SHARED_TEXT_SEGMENT 0x90000000U
515 #define GLOBAL_SHARED_DATA_SEGMENT 0xA0000000U
516 #define SHARED_TEXT_REGION_SIZE 0x10000000
517 #define SHARED_DATA_REGION_SIZE 0x10000000
518 
519  procCpuStats(stats);
520 
521  task_t a_task = mach_task_self();
522  task_basic_info_data_t ti;
523  mach_msg_type_number_t count = TASK_BASIC_INFO_COUNT;
524 
525  kern_return_t kr = task_info(a_task, TASK_BASIC_INFO, (task_info_t)&ti, &count);
526  if (kr != KERN_SUCCESS) {
527  dqm_error("[{0}] - Failed to get task_info: {1}", __FUNCTION__, mach_error_string(kr));
528  throw core::StatusCodeException(STATUS_CODE_FAILURE);
529  } else {
530  // resident size does not require any calculation. Virtual size
531  // needs to be adjusted if traversing memory objects do not include the
532  // globally shared text and data regions
533  mach_port_t object_name;
534  vm_address_t address;
535  vm_region_top_info_data_t info;
536 
537  vm_size_t rsize = ti.resident_size;
538  vm_size_t vsize = ti.virtual_size;
539  vm_size_t vprvt = 0;
540  vm_size_t size = 0;
541 
542  for (address = 0;; address += size) {
543  // get memory region
544  count = VM_REGION_TOP_INFO_COUNT;
545  if (vm_region(a_task, &address, &size, VM_REGION_TOP_INFO, (vm_region_info_t)&info, &count, &object_name) !=
546  KERN_SUCCESS) {
547  // no more memory regions.
548  break;
549  }
550 
551  if (address >= GLOBAL_SHARED_TEXT_SEGMENT &&
552  address < (GLOBAL_SHARED_DATA_SEGMENT + SHARED_DATA_REGION_SIZE)) {
553  // This region is private shared.
554  // Check if this process has the globally shared
555  // text and data regions mapped in. If so, adjust
556  // virtual memory size and exit loop.
557  if (info.share_mode == SM_EMPTY) {
558  vm_region_basic_info_data_64_t b_info;
559  count = VM_REGION_BASIC_INFO_COUNT_64;
560  if (vm_region_64(a_task, &address, &size, VM_REGION_BASIC_INFO, (vm_region_info_t)&b_info, &count,
561  &object_name) != KERN_SUCCESS) {
562  break;
563  }
564 
565  if (b_info.reserved) {
566  vsize -= (SHARED_TEXT_REGION_SIZE + SHARED_DATA_REGION_SIZE);
567  // break; // only for vsize
568  }
569  }
570  // Short circuit the loop if this isn't a shared
571  // private region, since that's the only region
572  // type we care about within the current address range.
573  if (info.share_mode != SM_PRIVATE) {
574  continue;
575  }
576  }
577  switch (info.share_mode) {
578  case SM_COW: {
579  if (info.ref_count == 1) {
580  vprvt += size;
581  } else {
582  vprvt += info.private_pages_resident * getpagesize();
583  }
584  break;
585  }
586  case SM_PRIVATE: {
587  vprvt += size;
588  break;
589  }
590  default:
591  break;
592  }
593  }
594 
595  stats.vm = (dqm_long)(rsize / 1024.);
596  stats.rss = (dqm_long)(vprvt / 1024.);
597  }
598  }

606  static void readDarwinNet(NetworkStats &stats) {
607  double unit = 1024.; // Store everything in kb
608 
609  // Get sizing info from sysctl and resize as needed.
610  int mib[] = {CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST, 0};
611  size_t currentSize{0};
612  if (sysctl(mib, 6, NULL, &currentSize, NULL, 0) != 0) {
613  const int errNum = errno;
614  dqm_error("[{0}] - Failed to get network buffer size: {1}", __FUNCTION__, errNum);
615  throw core::StatusCodeException(STATUS_CODE_FAILURE);
616  }
617  size_t sysctlBufferSize{0};
618  uint8_t *sysctlBuffer = nullptr;
619  if (!sysctlBuffer || (currentSize > sysctlBufferSize)) {
620  if (sysctlBuffer)
621  free(sysctlBuffer);
622  sysctlBufferSize = 0;
623  sysctlBuffer = (uint8_t *)malloc(currentSize);
624  if (!sysctlBuffer) {
625  const int errNum = errno;
626  dqm_error("[{0}] - Failed to get network statistics: {1}", __FUNCTION__, errNum);
627  throw core::StatusCodeException(STATUS_CODE_FAILURE);
628  }
629  sysctlBufferSize = currentSize;
630  }
631 
632  // Read in new data
633  if (sysctl(mib, 6, sysctlBuffer, &currentSize, NULL, 0) != 0) {
634  const int errNum = errno;
