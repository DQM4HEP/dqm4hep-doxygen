143  core::StatusCode GenericEventXMLReader::readNextEvent() {
144  EventPtr event = GenericEvent::make_shared();
145  GenericEvent *generic = event->getEvent<GenericEvent>();
146  // end of file !
147  if(nullptr == m_currentEvent) {
148  return STATUS_CODE_OUT_OF_RANGE;
149  }
150  TiXmlHandle handle(m_currentEvent);
151  int32_t eventType = static_cast<int32_t>(UNKNOWN_EVENT);
152  std::string source;
153  int32_t timeStamp = 0, eventNumber = 0, runNumber = 0;
154  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
155  "EventType", eventType));
156  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
157  "Source", source));
158  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
159  "TimeStamp", timeStamp));
160  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
161  "EventNumber", eventNumber));
162  RETURN_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, XmlHelper::readParameter(handle,
163  "RunNumber", runNumber));
164  event->setType(static_cast<EventType>(eventType));
165  event->setSource(source);
166  event->setTimeStamp(core::toTimePoint(timeStamp));
167  event->setEventNumber(eventNumber);
168  event->setRunNumber(runNumber);
169  // loop over <field> elements
170  for(auto field = m_currentEvent->FirstChildElement("field") ; nullptr != field ; field = field->NextSiblingElement("field")) {
171  std::string name, type;
172  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(field, "name", name));
173  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, XmlHelper::getAttribute(field, "type", type));
174  if("int" == type) {
175  if(nullptr == field->GetText()) {
176  return STATUS_CODE_FAILURE;
177  }
178  std::string content = field->GetText();
179  IntVector values;
180  core::tokenize(content, values, " ");
181  generic->setValues(name, values);
182  }
183  else if("float" == type) {
184  if(nullptr == field->GetText()) {
185  return STATUS_CODE_FAILURE;
186  }
187  std::string content = field->GetText();
188  FloatVector values;
189  core::tokenize(content, values, " ");
190  generic->setValues(name, values);
191  }
192  else if("double" == type) {
193  if(nullptr == field->GetText()) {
194  return STATUS_CODE_FAILURE;
195  }
196  std::string content = field->GetText();
197  DoubleVector values;
198  core::tokenize(content, values, " ");
199  generic->setValues(name, values);
200  }
201  else if("string" == type) {
202  StringVector values;
203  // Wrap string into <str> </str> elements to avoid string tokenize
204  for(auto strElement = field->FirstChildElement("str") ; nullptr != strElement ; strElement = strElement->NextSiblingElement("str")) {
205  values.push_back(strElement->GetText() != nullptr ? strElement->GetText() : "");
206  }
207  generic->setValues(name, values);
208  }
209  else {
210  dqm_warning( "Unrecognized field type '{0}' ! Skipping ...", type );
211  continue;
212  }
213  }
214  onEventRead().emit(event);
215  m_currentEvent = m_currentEvent->NextSiblingElement("event");
216  return STATUS_CODE_SUCCESS;
217  }
