244 
245  if (mysql_query(m_pMySQL, uquery.c_str())) {
246  dqm_error("MySQL query failed : {0}", mysql_error(m_pMySQL));
247  return STATUS_CODE_FAILURE;

249 
250  MYSQL_RES *pMySQLResult = mysql_store_result(m_pMySQL);
251 
252  if (!pMySQLResult) {
253  dqm_error("MySQL store result failed : {0}", mysql_error(m_pMySQL));
254  return STATUS_CODE_FAILURE;
255  }
256 
257  MYSQL_ROW row = mysql_fetch_row(pMySQLResult);
258 
259  if (!dqm4hep::core::stringToType(row[0], result)) {
260  mysql_free_result(pMySQLResult);
261  return STATUS_CODE_FAILURE;
262  }

265 
266  return STATUS_CODE_SUCCESS;
